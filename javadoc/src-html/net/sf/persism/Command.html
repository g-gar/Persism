039        /**
040         * Updates the data object in the database.
041         *
042         * @param object data object to update.
043         * @return usually 1 to indicate rows changed via JDBC.
044         * @throws PersismException Indicating the upcoming robot uprising.
045         */
046        public int update(Object object) throws PersismException {
047            List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
048            if (primaryKeys.size() == 0) {
049                throw new PersismException("Cannot perform update. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
050            }
051    
052            PreparedStatement st = null;
053            try {
054    
055                String updateStatement = metaData.getUpdateStatement(object, connection);
056                if (updateStatement == null || updateStatement.trim().length() == 0) {
057                    log.warn("No properties changed. No update required for Object: " + object + " class: " + object.getClass().getName());
058                    return 0;
059                }
060    
061                st = connection.prepareStatement(updateStatement);
062    
063                // These keys should always be in sorted order.
064                Map<String, PropertyInfo> allProperties = metaData.getTableColumns(object.getClass(), connection);
065                Map<String, PropertyInfo> changedProperties;
066                if (object instanceof Persistable) {
067                    changedProperties = metaData.getChangedColumns((Persistable) object, connection);
068                } else {
069                    changedProperties = allProperties;
070                }
071    
072                List<Object> params = new ArrayList<Object>(primaryKeys.size());
073                for (String column : changedProperties.keySet()) {
074                    if (!primaryKeys.contains(column)) {
075                        Object value = allProperties.get(column).getter.invoke(object);
076    
077                        if (value != null) {
078                            if (value.getClass().isEnum()) {
079                                value = "" + value; // convert enum to string.
080                            }
081    
082                            if (value instanceof java.util.Date) {
083                                java.util.Date dt = (java.util.Date) value;
084                                value = new Timestamp(dt.getTime());
085                            }
086    
087                            if (value instanceof String) {
088                                // check width
089                                PropertyInfo propertyInfo = allProperties.get(column);
090                                String str = (String) value;
091                                if (str.length() > propertyInfo.length) {
092                                    str = str.substring(0, propertyInfo.length);
093                                    // todo Should Persism strict should throw this as an exception?
094                                    log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
095                                    value = str;
096                                }
097                            }
098                        }
099                        params.add(value);
100                    }
101                }
102    
103                for (String column : primaryKeys) {
104                    params.add(allProperties.get(column).getter.invoke(object));
105                }
106                Util.setParameters(st, params.toArray());
107                int ret = st.executeUpdate();
108                return ret;
109            } catch (Exception e) {
110                try {
111                    if (connection != null && !connection.getAutoCommit()) {
112                        connection.rollback();
113                    }
114                } catch (SQLException e1) {
115                    log.error(e1.getMessage(), e1);
116                }
117                throw new PersismException(e);
118    
119            } finally {
120                Util.cleanup(st, null);
121            }
122        }

124        /**
125         * Inserts the data object in the database.
126         *
127         * @param object the data object to insert.
128         * @return usually 1 to indicate rows changed via JDBC.
129         * @throws PersismException When planet of the apes starts happening.
130         */
131        public int insert(Object object) throws PersismException {
132            String insertStatement = metaData.getInsertStatement(object, connection);
133    
134            PreparedStatement st = null;
135            java.sql.ResultSet rs = null;
136    
137            try {
138                // These keys should always be in sorted order.
139                Map<String, PropertyInfo> properties = metaData.getTableColumns(object.getClass(), connection);
140                Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
141    
142                List<String> generatedKeys = new ArrayList<String>(4);
143                for (ColumnInfo column : columns.values()) {
144                    if (column.generated) {
145                        generatedKeys.add(column.columnName);
146                    }
147                }
148    
149                if (generatedKeys.size() > 0) {
150                    String[] keyArray = generatedKeys.toArray(new String[0]);
151                    st = connection.prepareStatement(insertStatement, keyArray);
152                } else {
153                    st = connection.prepareStatement(insertStatement);
154                }
155    
156                boolean tableHasDefaultColumnValues = false;
157    
158                List<Object> params = new ArrayList<Object>();
159                for (ColumnInfo column : columns.values()) {
160    
161                    PropertyInfo propertyInfo = properties.get(column.columnName);
162    
163                    // todo is propertyInfo null ever? I don't think so. We only include columns where we know the property.
164                    if (!column.generated) {
165    
166                        // TODO This condition is repeated 3 times. We need to rearrange this code.
167                        // See MetaData getInsertStatement - Maybe we should return a new Object type for InsertStatement
168                        if (column.hasDefault) {
169                            // Do not include if this column has a default and no value has been
170                            // set on it's associated property.
171                            if (properties.get(column.columnName).getter.invoke(object) == null) {
172                                tableHasDefaultColumnValues = true;
173                                continue;
174                            }
175                        }
176    
177    
178                        Object value = propertyInfo.getter.invoke(object);
179    
180                        if (value != null) {
181    
182                            if (value.getClass().isEnum()) {
183                                value = "" + value; // convert enum to string.
184                            }
185    
186                            // sql.Date is a subclass so this would be true
187                            if (value instanceof java.util.Date) {
188                                java.util.Date dt = (java.util.Date) value;
189                                value = new Timestamp(dt.getTime());
190                            }
191    
192    
193                            if (value instanceof String) {
194                                // check width
195                                String str = (String) value;
196                                if (str.length() > propertyInfo.length) {
197                                    // todo should Persism strict throw this as an exception?
198                                    str = str.substring(0, propertyInfo.length);
199                                    log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
200                                    value = str;
201                                }
202                            }
203                        }
204                        params.add(value);
205                    }
206                }
207    
208                // https://forums.oracle.com/forums/thread.jspa?threadID=879222
209                // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
210                //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
211                Util.setParameters(st, params.toArray());
212                int ret = st.executeUpdate();
213                if (log.isDebugEnabled()) {
214                    log.debug("insert ret: " + ret);
215                }
216                if (generatedKeys.size() > 0) {
217                    rs = st.getGeneratedKeys();
218                }
219    
220                // TODO for now we can only support a single auto inc - need to test out other possible generated columns
221                for (String column : generatedKeys) {
222                    if (rs.next()) {
223    
224                        Method setter = properties.get(column).setter;
225    
226                        if (setter != null) {
227                            // todo do we really need to type these? Maybe if the DB uses a GUID?
228                            Object value = metaData.getTypedValue(setter.getParameterTypes()[0], rs, 1);
229                            if (log.isDebugEnabled()) {
230                                log.debug(column + " generated " + value);
231                                log.debug(setter);
232                            }
233                            setter.invoke(object, value);
234    
235                        } else {
236                            log.warn("no setter found for column " + column);
237                        }
238                    }
239                }
240    
241                if (tableHasDefaultColumnValues) {
242                    if (query == null) {
243                        query = new Query(connection);
244                    }
245                    // Read the full object back to update any properties which had defaults
246                    query.read(object);
247                }
248    
249                return ret;
250            } catch (Exception e) {
251                try {
252                    if (connection != null && !connection.getAutoCommit()) {
253                        connection.rollback();
254                    }
255                } catch (SQLException e1) {
256                    log.error(e1.getMessage(), e1);
257                }
258    
259                throw new PersismException(e);
260            } finally {
261                Util.cleanup(st, rs);
262            }
263        }

266        /**
267         * Deletes the data object object from the database.
268         *
269         * @param object data object to delete
270         * @return usually 1 to indicate rows changed via JDBC.
271         * @throws PersismException Perhaps when asteroid 1999 RQ36 hits us?
272         */
273        public int delete(Object object) throws PersismException {
274    
275            List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
276            if (primaryKeys.size() == 0) {
277                throw new PersismException("Cannot perform delete. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
278            }
279    
280            PreparedStatement st = null;
281            try {
282                String deleteStatement = metaData.getDeleteStatement(object, connection);
283                st = connection.prepareStatement(deleteStatement);
284    
285                // These keys should always be in sorted order.
286                Map<String, PropertyInfo> columns = metaData.getTableColumns(object.getClass(), connection);
287    
288                List<Object> params = new ArrayList<Object>(primaryKeys.size());
289                for (String column : primaryKeys) {
290                    params.add(columns.get(column).getter.invoke(object));
291                }
292                Util.setParameters(st, params.toArray());
293                int ret = st.executeUpdate();
294                return ret;
295            } catch (Exception e) {
296                try {
297                    if (connection != null && !connection.getAutoCommit()) {
298                        connection.rollback();
299                    }
300                } catch (SQLException e1) {
301                    log.error(e1.getMessage(), e1);
302                }
303                throw new PersismException(e);
304    
305            } finally {
306                Util.cleanup(st, null);
307            }
308        }

310        /**
311         * Execute an arbitrary SQL statement.
312         *
313         * @param sql
314         * @param parameters
315         */
316        public void execute(String sql, Object... parameters) {
317    
318            Statement st = null;
319    
320            try {
321    
322                if (parameters.length == 0) {
323                    st = connection.createStatement();
324                    st.execute(sql);
325                } else {
326                    st = connection.prepareStatement(sql);
327                    PreparedStatement pst = (PreparedStatement) st;
328                    Util.setParameters(pst, parameters);
329                    pst.execute();
330                }
331    
332            } catch (SQLException e) {
333                throw new PersismException(e);
334            } finally {
335                Util.cleanup(st, null);
336            }
337        }
