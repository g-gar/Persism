054    /**
055     * Updates the data object in the database.
056     *
057     * @param object data object to update.
058     * @return usually 1 to indicate rows changed via JDBC.
059     * @throws PersismException Indicating the upcoming robot uprising.
060     */
061    public int update(Object object) throws PersismException {
062        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
063        if (primaryKeys.size() == 0) {
064            throw new PersismException("Cannot perform update. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
065        }
066
067        PreparedStatement st = null;
068        try {
069
070            String updateStatement = metaData.getUpdateStatement(object, connection);
071            if (updateStatement == null || updateStatement.trim().length() == 0) {
072                log.warn("No properties changed. No update required for Object: " + object + " class: " + object.getClass().getName());
073                return 0;
074            }
075
076            st = connection.prepareStatement(updateStatement);
077
078            // These keys should always be in sorted order.
079            Map<String, PropertyInfo> allProperties = metaData.getTableColumns(object.getClass(), connection);
080            Map<String, PropertyInfo> changedProperties;
081            if (object instanceof Persistable) {
082                changedProperties = metaData.getChangedColumns((Persistable) object, connection);
083            } else {
084                changedProperties = allProperties;
085            }
086
087            List<Object> params = new ArrayList<Object>(primaryKeys.size());
088            for (String column : changedProperties.keySet()) {
089                if (!primaryKeys.contains(column)) {
090                    Object value = allProperties.get(column).getter.invoke(object);
091
092                    if (value != null) {
093                        if (value.getClass().isEnum()) {
094                            value = "" + value; // convert enum to string.
095                        }
096
097                        if (value instanceof java.util.Date) {
098                            java.util.Date dt = (java.util.Date) value;
099                            value = new Timestamp(dt.getTime());
100                        }
101
102                        if (value instanceof String) {
103                            // check width
104                            PropertyInfo propertyInfo = allProperties.get(column);
105                            String str = (String) value;
106                            if (str.length() > propertyInfo.length) {
107                                str = str.substring(0, propertyInfo.length);
108                                // todo Should Persism strict should throw this as an exception?
109                                log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
110                                value = str;
111                            }
112                        }
113                    }
114                    params.add(value);
115                }
116            }
117
118            for (String column : primaryKeys) {
119                params.add(allProperties.get(column).getter.invoke(object));
120            }
121            Util.setParameters(st, params.toArray());
122            int ret = st.executeUpdate();
123            return ret;
124
125        } catch (Exception e) {
126            Util.rollback(connection);
127            throw new PersismException(e);
128
129        } finally {
130            Util.cleanup(st, null);
131        }
132    }

134    /**
135     * Inserts the data object in the database.
136     *
137     * @param object the data object to insert.
138     * @return usually 1 to indicate rows changed via JDBC.
139     * @throws PersismException When planet of the apes starts happening.
140     */
141    public int insert(Object object) throws PersismException {
142        String insertStatement = metaData.getInsertStatement(object, connection);
143
144        PreparedStatement st = null;
145        java.sql.ResultSet rs = null;
146
147        try {
148            // These keys should always be in sorted order.
149            Map<String, PropertyInfo> properties = metaData.getTableColumns(object.getClass(), connection);
150            Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
151
152            List<String> generatedKeys = new ArrayList<String>(4);
153            for (ColumnInfo column : columns.values()) {
154                if (column.generated) {
155                    generatedKeys.add(column.columnName);
156                }
157            }
158
159            if (generatedKeys.size() > 0) {
160                String[] keyArray = generatedKeys.toArray(new String[0]);
161                st = connection.prepareStatement(insertStatement, keyArray);
162            } else {
163                st = connection.prepareStatement(insertStatement);
164            }
165
166            boolean tableHasDefaultColumnValues = false;
167
168            List<Object> params = new ArrayList<Object>();
169            for (ColumnInfo column : columns.values()) {
170
171                PropertyInfo propertyInfo = properties.get(column.columnName);
172
173                // todo is propertyInfo null ever? I don't think so. We only include columns where we know the property.
174                if (!column.generated) {
175
176                    // TODO This condition is repeated 3 times. We need to rearrange this code.
177                    // See MetaData getInsertStatement - Maybe we should return a new Object type for InsertStatement
178                    if (column.hasDefault) {
179                        // Do not include if this column has a default and no value has been
180                        // set on it's associated property.
181                        if (properties.get(column.columnName).getter.invoke(object) == null) {
182                            tableHasDefaultColumnValues = true;
183                            continue;
184                        }
185                    }
186
187
188                    Object value = propertyInfo.getter.invoke(object);
189
190                    if (value != null) {
191
192                        if (value.getClass().isEnum()) {
193                            value = "" + value; // convert enum to string.
194                        }
195
196                        // sql.Date is a subclass so this would be true
197                        if (value instanceof java.util.Date) {
198                            java.util.Date dt = (java.util.Date) value;
199                            value = new Timestamp(dt.getTime());
200                        }
201
202
203                        if (value instanceof String) {
204                            // check width
205                            String str = (String) value;
206                            if (str.length() > propertyInfo.length) {
207                                // todo should Persism strict throw this as an exception?
208                                str = str.substring(0, propertyInfo.length);
209                                log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
210                                value = str;
211                            }
212                        }
213                    }
214                    params.add(value);
215                }
216            }
217
218            // https://forums.oracle.com/forums/thread.jspa?threadID=879222
219            // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
220            //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
221            Util.setParameters(st, params.toArray());
222            int ret = st.executeUpdate();
223            if (log.isDebugEnabled()) {
224                log.debug("insert ret: " + ret);
225            }
226            if (generatedKeys.size() > 0) {
227                rs = st.getGeneratedKeys();
228            }
229
230            // TODO for now we can only support a single auto inc - need to test out other possible generated columns
231            for (String column : generatedKeys) {
232                if (rs.next()) {
233
234                    Method setter = properties.get(column).setter;
235
236                    if (setter != null) {
237                        // todo do we really need to type these? Maybe if the DB uses a GUID?
238                        Object value = metaData.getTypedValue(setter.getParameterTypes()[0], rs, 1);
239                        if (log.isDebugEnabled()) {
240                            log.debug(column + " generated " + value);
241                            log.debug(setter);
242                        }
243                        setter.invoke(object, value);
244
245                    } else {
246                        log.warn("no setter found for column " + column);
247                    }
248                }
249            }
250
251            if (tableHasDefaultColumnValues) {
252                // Read the full object back to update any properties which had defaults
253                fetch(object);
254            }
255
256            return ret;
257        } catch (Exception e) {
258            Util.rollback(connection);
259            throw new PersismException(e);
260        } finally {
261            Util.cleanup(st, rs);
262        }
263    }

334    /**
335     * Query for a list of objects of the specified class using the specified SQL query and parameters.
336     * The type of the list can be Data Objects or native Java Objects.
337     *
338     * @param objectClass class of objects to return.s
339     * @param sql         query string to execute.
340     * @param parameters  parameters to the query.
341     * @param <T>
342     * @return a list of objects of the specified class using the specified SQL query and parameters.
343     * @throws PersismException
344     */
345    public <T> List<T> query(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
346        List<T> list = new ArrayList<T>(32);
347
348        Result result = new Result();
349
350        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
351        boolean readPrimitive = Types.getType(objectClass) != null;
352
353        if (!readPrimitive && objectClass.getAnnotation(QueryResult.class) == null) {
354            metaData.getTableColumns(objectClass, connection); // TODO Make sure columns are initialized properly if this is a table WHY?
355        }
356
357        try {
358
359            exec(result, sql, parameters);
360
361            while (result.rs.next()) {
362
363                if (readPrimitive) {
364                    list.add((T) readPrimitive(result.rs, 1, objectClass));
365                } else {
366                    // should be getDeclaredConstructor().newInstance() now.
367                    T t = objectClass.newInstance();
368                    t = (T) readObject(t, result.rs);
369                    list.add(t);
370                }
371            }
372
373        } catch (IllegalAccessException | InstantiationException | InvocationTargetException | IOException e) {
374            throw new PersismException(e);
375        } catch (SQLException e) {
376            Util.rollback(connection);
377            throw new PersismException(e);
378        } finally {
379            Util.cleanup(result.st, result.rs);
380        }
381
382        return list;
383
384    }

386    /**
387     * Fetch an object from the database by it's primary key.
388     * You should instantiate the object and set the primary key properties before calling this method.
389     *
390     * @param object Data object to read from the database.
391     * @return true if the object was found by the primary key.
392     * @throws PersismException if something goes wrong.
393     */
394    public boolean fetch(Object object) throws PersismException {
395        Class objectClass = object.getClass();
396
397        // If we know this type it means it's a primitive type. This method cannot be used for primitives
398        boolean readPrimitive = Types.getType(objectClass) != null;
399        if (readPrimitive) {
400            throw new PersismException("Cannot read a primitive type object with this method.");
401        }
402
403        List<String> primaryKeys = metaData.getPrimaryKeys(objectClass, connection);
404        if (primaryKeys.size() == 0) {
405            throw new PersismException("Cannot perform readObjectByPrimary. " + metaData.getTableName(objectClass) + " has no primary keys.");
406        }
407
408        Map<String, PropertyInfo> properties = metaData.getTableColumns(object.getClass(), connection);
409        List<Object> params = new ArrayList<>(primaryKeys.size());
410
411        Result result = new Result();
412        try {
413            for (String column : primaryKeys) {
414                PropertyInfo propertyInfo = properties.get(column);
415                params.add(propertyInfo.getter.invoke(object));
416            }
417
418            exec(result, metaData.getSelectStatement(object, connection), params.toArray());
419
420            if (result.rs.next()) {
421                readObject(object, result.rs);
422                return true;
423            }
424            return false;
425
426        } catch (IllegalAccessException | InvocationTargetException | IOException e) {
427            throw new PersismException(e);
428        } catch (SQLException e) {
429            Util.rollback(connection);
430            throw new PersismException(e);
431        } finally {
432            Util.cleanup(result.st, result.rs);
433        }
434    }

470    public <T> T fetch(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
471        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
472        boolean readPrimitive = Types.getType(objectClass) != null;
473
474        if (!readPrimitive && objectClass.getAnnotation(QueryResult.class) == null) {
475            // Make sure columns are initialized properly if this is a table  todo  why?
476            metaData.getTableColumns(objectClass, connection);
477        }
478
479        Result result = new Result();
480        try {
481
482            exec(result, sql, parameters);
483
484            if (result.rs.next()) {
485
486                if (readPrimitive) {
487                    return (T) readPrimitive(result.rs, 1, objectClass);

489                } else {

517    List<LinkedHashMap<String, Object>> query(String sql, Object... parameters) {

520        ResultSet rs = null;
521
522        List<LinkedHashMap<String, Object>> results = new ArrayList<LinkedHashMap<String, Object>>(32);
523        try {
524

526                st = connection.createStatement();
527                rs = st.executeQuery(sql);
528            } else {
529                st = connection.prepareStatement(sql);
530
531                PreparedStatement pst = (PreparedStatement) st;
532                int n = 1;
533                for (Object o : parameters) {
534                    pst.setObject(n, o);
535                    n++;
536                }
537                rs = pst.executeQuery();
538            }
539
540
541            while (rs.next()) {
542                LinkedHashMap<String, Object> map = new LinkedHashMap<String, Object>(32);
543                int n = rs.getMetaData().getColumnCount();
544                for (int j = 1; j <= n; j++) {
545                    map.put(rs.getMetaData().getColumnLabel(j), rs.getObject(j));
546                }
547                results.add(map);
548            }
549        } catch (SQLException e) {
550            throw new PersismException(e);
551        } finally {
552            Util.cleanup(st, rs);
553        }
554
555        return results;

569    private Result exec(Result result, String sql, Object... parameters) throws SQLException {
570        if (sql.toLowerCase().contains("select ")) {
571            result.st = connection.prepareStatement(sql);
572
573            PreparedStatement pst = (PreparedStatement) result.st;
574            Util.setParameters(pst, parameters);
575            result.rs = pst.executeQuery();
576        } else {
577            // todo unit tests need to cover this.
578            result.st = connection.prepareCall("{call " + sql + "}");
579
580            CallableStatement cst = (CallableStatement) result.st;
581            Util.setParameters(cst, parameters);
582            result.rs = cst.executeQuery();
583        }
584        return result;
585    }

587    private <T> T readObject(Object object, ResultSet rs) throws IllegalAccessException, SQLException, InvocationTargetException, IOException {
588
589        Class objectClass = object.getClass();
590        // We should never call this method with a primitive type.
591        assert Types.getType(objectClass) == null;
592
593        Map<String, PropertyInfo> properties;
594        if (objectClass.getAnnotation(QueryResult.class) == null) {
595            properties = metaData.getTableColumns(objectClass, connection);
596        } else {
597            properties = metaData.getQueryColumns(objectClass, rs);
598        }
599
600        // Test if all properties have column mapping and throw PersismException if not
601        // This block verifies that the object is fully initialized.
602        // Any properties not marked by NotMapped should have been set (or if they have a getter only)
603        // If not throw a PersismException
604        Collection<PropertyInfo> allProperties = MetaData.getPropertyInfo(objectClass);
605        if (properties.values().size() < allProperties.size()) {
606            Set<PropertyInfo> missing = new HashSet<PropertyInfo>(allProperties.size());
607            missing.addAll(allProperties);
608            missing.removeAll(properties.values());
609
610            StringBuilder sb = new StringBuilder();
611            String sep = "";
612            for (PropertyInfo prop : missing) {
613                sb.append(sep).append(prop.propertyName);
614                sep = ",";
615            }
616
617            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not found in the queried columns (" + sb + ").");
618        }
619
620
621        ResultSetMetaData rsmd = rs.getMetaData();
622        int columnCount = rsmd.getColumnCount();
623        List<String> foundColumns = new ArrayList<String>(columnCount);
624
625        for (int j = 1; j <= columnCount; j++) {
626
627            String columnName = rsmd.getColumnLabel(j);
628            PropertyInfo columnProperty = properties.get(columnName);
629
630            if (columnProperty != null) {
631                Class getterType = columnProperty.getter.getReturnType();
632
633                Object value = readPrimitive(rs, j, getterType);
634
635                foundColumns.add(columnName);
636
637                if (value != null) {
638                    try {
639                        columnProperty.setter.invoke(object, value);
640                    } catch (IllegalArgumentException e) {
641                        String msg = e.getMessage() + " Object " + objectClass + ". Column: " + columnName + " Type of property: " + getterType + " - Type read: " + value.getClass() + " VALUE: " + value;
642                        throw new PersismException(msg, e);
643                    }
644
645                }
646            }
647        }
648
649        if (foundColumns.size() < properties.keySet().size()) {
650
651            Set<String> missing = new HashSet<String>(columnCount);
652            missing.addAll(properties.keySet());
653            missing.removeAll(foundColumns);
654
655            // todo maybe strict mode off logs warn? Should we do this if this is Query vs Table?
656            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not found in the queried columns. : " + missing);
657        }
658
659        if (object instanceof Persistable) {
660            // Save this object's initial state to later detect changed properties
661            ((Persistable) object).saveReadState();
662        }
663
664        return (T) object;
665
666    }

668    private Object readPrimitive(ResultSet rs, int column, Class returnType) throws SQLException, IOException {
669        ResultSetMetaData resultSetMetaData = rs.getMetaData();
670        int sqlColumnType = resultSetMetaData.getColumnType(column);
671
672        Types columnType = Types.convert(sqlColumnType); // note this could be null if we can't match a type
673        // Since there is no specific SQL column type for UUID we will use the return type to detect it.
674        if (returnType.equals(UUID.class)) {
675            // Check the return type for UUID since resultSetMetaData.getColumnType(column) has no UUID type
676            // it always returns a char or nvarchar so we'll just test and set it here. FFS.
677            columnType = Types.UUIDType;
678        }
679        String columnName = resultSetMetaData.getColumnLabel(column);
680
681        Object value;
682        if (columnType != null) {
683            switch (columnType) {
684
685                case TimestampType:
686                    // work around to Oracle reading a oracle.sql.TIMESTAMP class with getObject
687                    value = rs.getTimestamp(column);
688                    break;
689                case ClobType:
690                    value = rs.getClob(column);
691                    InputStream in = ((Clob) value).getAsciiStream();
692                    StringWriter write = new StringWriter();
693
694                    int c = -1;
695                    while ((c = in.read()) != -1) {
696                        write.write(c);
697                    }
698                    write.flush();
699                    value = write.toString();
700                    break;
701                case BlobType:
702                    // todo BlobType
703                case InputStreamType:
704                    // todo InputStreamType
705                case ReaderType:
706                    // todo ReaderType
707                case EnumType:
708                    // todo EnumType?
709                case UUIDType:
710                    value = rs.getObject(column);
711                    if (value != null) {
712                        value = UUID.fromString("" + value);
713                    }
714                    break;
715                case floatType:
716                case FloatType:
717                case doubleType:
718                case DoubleType:
719                    if (returnType.equals(java.math.BigDecimal.class)) {
720                        value = rs.getBigDecimal(column);
721                    } else if (returnType.equals(java.lang.Double.class)) {
722                        value = rs.getDouble(column);
723                    } else {
724                        value = rs.getFloat(column);
725                    }
726                    break;
727
728                default:
729                    value = rs.getObject(column);
730            }
731
732        } else {
733            log.warn("Column type not known for SQL type " + sqlColumnType);
734            value = rs.getObject(column);
735        }
736
737        // If value is null or column type is unknown - no need to try to convert anything.
738        if (value != null && columnType != null) {
739
740            Types valueType = Types.getType(value.getClass());
741
742            // try to convert or cast the value to the proper type.
743            // todo do code coverage for each specific type
744            switch (valueType) {
745
746                case booleanType:
747                case BooleanType:
748                    break;
749
750                case byteType:
751                case ByteType:
752                case shortType:
753                case ShortType:
754                case integerType:
755                case IntegerType:
756                    // int to bool
757                    if (returnType == Boolean.class || returnType == boolean.class) {
758                        value = (Integer.valueOf("" + value) == 0) ? false : true;
759                    }
760                    break;
761
762                case longType:
763                case LongType:
764                    // long to date
765                    if (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
766                        long lval = Long.valueOf("" + value);
767
768                        if (returnType.equals(java.sql.Date.class)) {
769                            value = new java.sql.Date(lval);
770                        } else {
771                            value = new java.util.Date(lval);
772                        }
773                    } else if (returnType == Integer.class || returnType == int.class) {
774                        log.warn("Possible overflow column " + columnName + " - Property is INT and column value is LONG");
775                        value = Integer.parseInt("" + value);
776                    }
777
778                    break;
779
780                case floatType:
781                case FloatType:
782                    break;
783
784                case doubleType:
785                case DoubleType:
786                    // float or doubles to BigDecimal
787                    if (returnType == BigDecimal.class) {
788                        value = new BigDecimal("" + value);
789                    } else if (returnType == Float.class || returnType == float.class) {
790                        // todo add tests for this
791                        log.warn("Possible overflow column " + columnName + " - Property is FLOAT and column value is DOUBLE");
792                        value = Float.parseFloat("" + value);
793                    } else if (returnType == Integer.class || returnType == int.class) {
794                        log.warn("Possible overflow column " + columnName + " - Property is INT and column value is DOUBLE");
795                        String val = "" + value;
796                        if (val.contains(".")) {
797                            val = val.substring(0, val.indexOf("."));
798                        }
799                        value = Integer.parseInt(val);
800                    }
801                    break;
802
803                case BigDecimalType:
804                    // mostly oracle
805                    if (returnType == Float.class || returnType == float.class) {
806                        value = ((BigDecimal) value).floatValue();
807                    } else if (returnType == Double.class || returnType == double.class) {
808                        value = ((BigDecimal) value).doubleValue();
809                    } else if (returnType == Long.class || returnType == long.class) {
810                        value = ((BigDecimal) value).longValue();
811                    } else if (returnType == Integer.class || returnType == int.class) {
812                        value = ((BigDecimal) value).intValue();
813                    } else if (returnType == Boolean.class || returnType == boolean.class) {
814                        value = ((BigDecimal) value).intValue() == 1;
815                    }
816                    break;
817
818                case StringType:
819
820                    // Read a string but we want a date
821                    if (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
822                        // This condition occurs in SQLite when you have a datetime with default annotated
823                        // the format returned is 2012-06-02 19:59:49
824                        Date dval = null;
825                        try {
826                            // Used for SQLite returning dates as Strings under some conditions
827                            DateFormat df = new SimpleDateFormat("yyyy-MM-DD hh:mm:ss");
828                            dval = df.parse("" + value);
829                        } catch (ParseException e) {
830                            String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + returnType + " - Type read: " + value.getClass() + " VALUE: " + value;
831                            throw new PersismException(msg, e);
832                        }
833
834                        if (returnType.equals(java.sql.Date.class)) {
835                            value = new java.sql.Date(dval.getTime());
836                        } else {
837                            value = dval;
838                        }
839
840                    } else if (returnType.isEnum()) {
841                        // If this is an enum do a case insensitive comparison
842                        Object[] enumConstants = returnType.getEnumConstants();
843                        for (Object element : enumConstants) {
844                            if (("" + value).equalsIgnoreCase(element.toString())) {
845                                value = element;
846                                break;
847                            }
848                        }
849                    }
850
851                    break;
852
853                case characterType:
854                case CharacterType:
855                    break;
856
857                case UtilDateType:
858                    break;
859                case SQLDateType:
860                    break;
861                case TimeType:
862                    break;
863
864                case TimestampType:
865                    if (returnType.isAssignableFrom(Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
866                        if (returnType.equals(java.sql.Date.class)) {
867                            value = new java.sql.Date(((Timestamp) value).getTime());
868                        } else {
869                            value = new Date(((Timestamp) value).getTime());
870                        }
871                    } else {
872                        value = ((Timestamp) value).getTime();
873                    }
874
875                    break;
876
877                case byteArrayType:
878                    break;
879                case ByteArrayType:
880                    break;
881                case charArrayType:
882                    break;
883                case CharArrayType:
884                    break;
885                case ClobType:
886                    // Convert to string
887                    if (value != null) {
888                        value = "" + value;
889                    }
890                    break;
891                case BlobType:
892                    break;
893                case InputStreamType:
894                    break;
895                case ReaderType:
896                    break;
897                case EnumType:
898                    break;
899            }
900        }
901
902        return value;
903    }
