046    /**
047     * Updates the data object in the database.
048     * @deprecated
049     * @see net.sf.persism.Session#update(Object)
050     * @param object data object to update.
051     * @return usually 1 to indicate rows changed via JDBC.
052     * @throws PersismException Indicating the upcoming robot uprising.
053     */
054    public int update(Object object) throws PersismException {
055        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
056        if (primaryKeys.size() == 0) {
057            throw new PersismException("Cannot perform update. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
058        }
059
060        PreparedStatement st = null;
061        try {
062
063            String updateStatement = metaData.getUpdateStatement(object, connection);
064            if (updateStatement == null || updateStatement.trim().length() == 0) {
065                log.warn("No properties changed. No update required for Object: " + object + " class: " + object.getClass().getName());
066                return 0;
067            }
068
069            st = connection.prepareStatement(updateStatement);
070
071            // These keys should always be in sorted order.
072            Map<String, PropertyInfo> allProperties = metaData.getTableColumns(object.getClass(), connection);
073            Map<String, PropertyInfo> changedProperties;
074            if (object instanceof Persistable) {
075                changedProperties = metaData.getChangedColumns((Persistable) object, connection);
076            } else {
077                changedProperties = allProperties;
078            }
079
080            List<Object> params = new ArrayList<Object>(primaryKeys.size());
081            for (String column : changedProperties.keySet()) {
082                if (!primaryKeys.contains(column)) {
083                    Object value = allProperties.get(column).getter.invoke(object);
084
085                    if (value != null) {
086                        if (value.getClass().isEnum()) {
087                            value = "" + value; // convert enum to string.
088                        }
089
090                        if (value instanceof java.util.Date) {
091                            java.util.Date dt = (java.util.Date) value;
092                            value = new Timestamp(dt.getTime());
093                        }
094
095                        if (value instanceof String) {
096                            // check width
097                            PropertyInfo propertyInfo = allProperties.get(column);
098                            String str = (String) value;
099                            if (str.length() > propertyInfo.length) {
100                                str = str.substring(0, propertyInfo.length);
101                                // todo Should Persism strict should throw this as an exception?
102                                log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
103                                value = str;
104                            }
105                        }
106                    }
107                    params.add(value);
108                }
109            }
110
111            for (String column : primaryKeys) {
112                params.add(allProperties.get(column).getter.invoke(object));
113            }
114            Util.setParameters(st, params.toArray());
115            int ret = st.executeUpdate();
116            return ret;
117
118        } catch (Exception e) {
119            Util.rollback(connection);
120            throw new PersismException(e);
121
122        } finally {
123            Util.cleanup(st, null);
124        }
125    }

127    /**
128     * Inserts the data object in the database.
129     * @deprecated
130     * @see net.sf.persism.Session#insert(Object)
131     * @param object the data object to insert.
132     * @return usually 1 to indicate rows changed via JDBC.
133     * @throws PersismException When planet of the apes starts happening.
134     */
135    public int insert(Object object) throws PersismException {
136        String insertStatement = metaData.getInsertStatement(object, connection);
137
138        PreparedStatement st = null;
139        java.sql.ResultSet rs = null;
140
141        try {
142            // These keys should always be in sorted order.
143            Map<String, PropertyInfo> properties = metaData.getTableColumns(object.getClass(), connection);
144            Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
145
146            List<String> generatedKeys = new ArrayList<String>(4);
147            for (ColumnInfo column : columns.values()) {
148                if (column.generated) {
149                    generatedKeys.add(column.columnName);
150                }
151            }
152
153            if (generatedKeys.size() > 0) {
154                String[] keyArray = generatedKeys.toArray(new String[0]);
155                st = connection.prepareStatement(insertStatement, keyArray);
156            } else {
157                st = connection.prepareStatement(insertStatement);
158            }
159
160            boolean tableHasDefaultColumnValues = false;
161
162            List<Object> params = new ArrayList<Object>();
163            for (ColumnInfo column : columns.values()) {
164
165                PropertyInfo propertyInfo = properties.get(column.columnName);
166
167                // todo is propertyInfo null ever? I don't think so. We only include columns where we know the property.
168                if (!column.generated) {
169
170                    // TODO This condition is repeated 3 times. We need to rearrange this code.
171                    // See MetaData getInsertStatement - Maybe we should return a new Object type for InsertStatement
172                    if (column.hasDefault) {
173                        // Do not include if this column has a default and no value has been
174                        // set on it's associated property.
175                        if (properties.get(column.columnName).getter.invoke(object) == null) {
176                            tableHasDefaultColumnValues = true;
177                            continue;
178                        }
179                    }
180
181
182                    Object value = propertyInfo.getter.invoke(object);
183
184                    if (value != null) {
185
186                        if (value.getClass().isEnum()) {
187                            value = "" + value; // convert enum to string.
188                        }
189
190                        // sql.Date is a subclass so this would be true
191                        if (value instanceof java.util.Date) {
192                            java.util.Date dt = (java.util.Date) value;
193                            value = new Timestamp(dt.getTime());
194                        }
195
196
197                        if (value instanceof String) {
198                            // check width
199                            String str = (String) value;
200                            if (str.length() > propertyInfo.length) {
201                                // todo should Persism strict throw this as an exception?
202                                str = str.substring(0, propertyInfo.length);
203                                log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
204                                value = str;
205                            }
206                        }
207                    }
208                    params.add(value);
209                }
210            }
211
212            // https://forums.oracle.com/forums/thread.jspa?threadID=879222
213            // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
214            //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
215            Util.setParameters(st, params.toArray());
216            int ret = st.executeUpdate();
217            if (log.isDebugEnabled()) {
218                log.debug("insert ret: " + ret);
219            }
220            if (generatedKeys.size() > 0) {
221                rs = st.getGeneratedKeys();
222            }
223
224            // TODO for now we can only support a single auto inc - need to test out other possible generated columns
225            for (String column : generatedKeys) {
226                if (rs.next()) {
227
228                    Method setter = properties.get(column).setter;
229
230                    if (setter != null) {
231                        // todo do we really need to type these? Maybe if the DB uses a GUID?
232                        Object value = metaData.getTypedValue(setter.getParameterTypes()[0], rs, 1);
233                        if (log.isDebugEnabled()) {
234                            log.debug(column + " generated " + value);
235                            log.debug(setter);
236                        }
237                        setter.invoke(object, value);
238
239                    } else {
240                        log.warn("no setter found for column " + column);
241                    }
242                }
243            }
244
245            if (tableHasDefaultColumnValues) {
246                if (query == null) {
247                    query = new Query(connection);
248                }
249                // Read the full object back to update any properties which had defaults
250                query.read(object);
251            }
252
253            return ret;
254        } catch (Exception e) {
255            Util.rollback(connection);
256            throw new PersismException(e);
257        } finally {
258            Util.cleanup(st, rs);
259        }
260    }

263    /**
264     * Deletes the data object object from the database.
265     * @deprecated
266     * @see net.sf.persism.Session#delete(Object)
267     * @param object data object to delete
268     * @return usually 1 to indicate rows changed via JDBC.
269     * @throws PersismException Perhaps when asteroid 1999 RQ36 hits us?
270     */
271    public int delete(Object object) throws PersismException {
272
273        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
274        if (primaryKeys.size() == 0) {
275            throw new PersismException("Cannot perform delete. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
276        }
277
278        PreparedStatement st = null;
279        try {
280            String deleteStatement = metaData.getDeleteStatement(object, connection);
281            st = connection.prepareStatement(deleteStatement);
282
283            // These keys should always be in sorted order.
284            Map<String, PropertyInfo> columns = metaData.getTableColumns(object.getClass(), connection);
285
286            List<Object> params = new ArrayList<Object>(primaryKeys.size());
287            for (String column : primaryKeys) {
288                params.add(columns.get(column).getter.invoke(object));
289            }
290            Util.setParameters(st, params.toArray());
291            int ret = st.executeUpdate();
292            return ret;
293
294        } catch (Exception e) {
295            Util.rollback(connection);
296            throw new PersismException(e);
297
298        } finally {
299            Util.cleanup(st, null);
300        }
301    }

303    /**
304     * Execute an arbitrary SQL statement.
305     * @deprecated
306     * @see net.sf.persism.Session#execute(String, Object...)
307     * @param sql
308     * @param parameters
309     */
310    public void execute(String sql, Object... parameters) {
311
312        Statement st = null;
313
314        try {
315
316            if (parameters.length == 0) {
317                st = connection.createStatement();
318                st.execute(sql);
319            } else {
320                st = connection.prepareStatement(sql);
321                PreparedStatement pst = (PreparedStatement) st;
322                Util.setParameters(pst, parameters);
323                pst.execute();
324            }
325
326        } catch (SQLException e) {
327            throw new PersismException(e);
328        } finally {
329            Util.cleanup(st, null);
330        }
331    }

335     * Execute an arbut? Why bother? Why even put in Command?
336     * @param sql
337     * @param parameters
338     * @return
339     */
340     List<LinkedHashMap<String, Object>> query(String sql, Object... parameters) {
341
342        Statement st = null;
343        ResultSet rs = null;
344
345        List<LinkedHashMap<String, Object>> results = new ArrayList<LinkedHashMap<String, Object>>(32);
346        try {
347
348            if (parameters.length == 0) {
349                st = connection.createStatement();
350                rs = st.executeQuery(sql);
351            } else {
352                st = connection.prepareStatement(sql);
353
354                PreparedStatement pst = (PreparedStatement) st;
355                int n = 1;
356                for (Object o : parameters) {
357                    pst.setObject(n, o);
358                    n++;
359                }
360                rs = pst.executeQuery();
361            }
362
363
364            while (rs.next()) {
365                LinkedHashMap<String, Object> map = new LinkedHashMap<String, Object>(32);
366                int n = rs.getMetaData().getColumnCount();
367                for (int j = 1; j <=n; j++) {
368                    map.put(rs.getMetaData().getColumnLabel(j), rs.getObject(j));
369                }
370                results.add(map);
371            }
372        } catch (SQLException e) {
373            throw new PersismException(e);
374        } finally {
375            Util.cleanup(st, rs);
376        }
377
378        return results;
379    }
