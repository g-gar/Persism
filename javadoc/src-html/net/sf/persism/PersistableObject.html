001package net.sf.persism;
002
003import java.lang.reflect.InvocationTargetException;
004import java.util.Collection;
005
006/**
007 * Data objects can optionally inherit from this class. Persism uses information from this class to detect which
008 * properties are modified in your data objects and only includes those columns in the SQL UPDATE statements.
009 *
010 * @author Dan Howard
011 * @since 9/15/11 7:14 AM
012 */
013public abstract class PersistableObject<T> implements Persistable<T> {
014
015    private T originalValue = null;
016
017    public final void saveReadState() throws PersismException {
018        originalValue = clone();
019    }
020
021    public final T getOriginalValue() {
022        return originalValue;
023    }
024
025    /**
026     * Used for getting originalValue
027     * @return Clone of T
028     */
029    @Override
030    public final T clone()  {
031        try {
032            return (T)super.clone();
033        } catch (CloneNotSupportedException e) {
034            throw new PersismException(e.getMessage(), e);
035        }
036    }
037}
