013import java.lang.annotation.ElementType;
014import java.lang.annotation.Retention;
015import java.lang.annotation.RetentionPolicy;
016import java.lang.annotation.Target;
017
018/**
019 * Defines the table name for a given class. This annotation is optional since Persism can detect the table name
020 * from the class in most situations.
021 */
022@Retention(value = RetentionPolicy.RUNTIME)
023@Target(ElementType.TYPE)
024public @interface TableName {
025
026    String value();
027
028}
