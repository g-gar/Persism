050        /**
051         * Read a list of objects of the specified class using the specified SQL query and parameters.
052         * The type of the list can be Data Objects or native Java Objects.
053         *
054         * @param objectClass class of objects to return.
055         * @param sql         query string to execute.
056         * @param parameters  parameters to the query.
057         * @param <T>
058         * @return a list of objects of the specified class using the specified SQL query and parameters.
059         * @throws PersismException
060         */
061        public <T> List<T> readList(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
062            List<T> list = new ArrayList<T>(32);
063    
064            Result result = new Result();
065    
066            // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
067            boolean readPrimitive = Types.getType(objectClass) != null;
068    
069            if (!readPrimitive && objectClass.getAnnotation(QueryResult.class) == null) {
070                metaData.getTableColumns(objectClass, connection); // TODO Make sure columns are initialized properly if this is a table WHY?
071            }
072    
073            try {
074    
075                exec(result, sql, parameters);
076    
077                while (result.rs.next()) {
078    
079                    if (readPrimitive) {
080                        list.add((T) readPrimitive(result.rs, 1, objectClass));
081                    } else {
082                        // should be getDeclaredConstructor().newInstance() now.
083                        T t = objectClass.newInstance();
084                        t = (T) readObject(t, result.rs);
085                        list.add(t);
086                    }
087                }
088    
089            } catch (IllegalAccessException e) {
090                throw new PersismException(e);
091            } catch (InstantiationException e) {
092                throw new PersismException(e);
093            } catch (InvocationTargetException e) {
094                throw new PersismException(e);
095            } catch (SQLException e) {
096                // todo check transaction and rollback. Even though it's a query a transaction may exist. SEE OTHER PLACES WHERE THIS COULD OCCUR
097                throw new PersismException(e);
098            } catch (IOException e) {
099                throw new PersismException(e);
100            } finally {
101                Util.cleanup(result.st, result.rs);
102            }
103    
104            return list;
105        }

111        /**
112         * Read an object from the database by it's primary key.
113         * You should instantiate the object and set the primary key properties before calling this method.
114         *
115         * @param object Data object to read from the database.
116         * @return true if the object was found by the primary key.
117         * @throws PersismException if something goes wrong.
118         */
119        public boolean read(Object object) throws PersismException {
120    
121            Class objectClass = object.getClass();
122    
123            // If we know this type it means it's a primitive type. This method cannot be used for primitives
124            boolean readPrimitive = Types.getType(objectClass) != null;
125            if (readPrimitive) {
126                throw new PersismException("Cannot read a primitive type object with this method.");
127            }
128    
129            List<String> primaryKeys = metaData.getPrimaryKeys(objectClass, connection);
130            if (primaryKeys.size() == 0) {
131                throw new PersismException("Cannot perform readObjectByPrimary. " + metaData.getTableName(objectClass) + " has no primary keys.");
132            }
133    
134            Map<String, PropertyInfo> properties = metaData.getTableColumns(object.getClass(), connection);
135            List<Object> params = new ArrayList<Object>(primaryKeys.size());
136    
137            Result result = new Result();
138            try {
139                for (String column : primaryKeys) {
140                    PropertyInfo propertyInfo = properties.get(column);
141                    params.add(propertyInfo.getter.invoke(object));
142                }
143    
144                exec(result, metaData.getSelectStatement(object, connection), params.toArray());
145    
146                if (result.rs.next()) {
147                    readObject(object, result.rs);
148                    return true;
149                }
150                return false;
151    
152            } catch (IllegalAccessException e) {
153                throw new PersismException(e);
154            } catch (InvocationTargetException e) {
155                throw new PersismException(e);
156            } catch (SQLException e) {
157                throw new PersismException(e);
158            } catch (IOException e) {
159                throw new PersismException(e);
160            } finally {
161                Util.cleanup(result.st, result.rs);
162            }
163        }

165        /**
166         * Reads an object of the specified type from the database. The type can a be Data Object or a native Java Object.
167         *
168         * @param objectClass Type of returned value
169         * @param sql         query - this would usually be a select OR a select of a single column if the type is a primitive.
170         *                    If this is a primitive type then this method will only look at the 1st column in the result.
171         * @param parameters  parameters to the query.
172         * @param <T>
173         * @return value read from the database
174         * @throws PersismException Well, this is a runtime exception so actually it could be anything really.
175         */
176        public <T> T read(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
177    
178            // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
179            boolean readPrimitive = Types.getType(objectClass) != null;
180    
181            if (!readPrimitive && objectClass.getAnnotation(QueryResult.class) == null) {
182                // Make sure columns are initialized properly if this is a table  todo  why?
183                metaData.getTableColumns(objectClass, connection);
184            }
185    
186            Result result = new Result();
187            try {
188    
189                exec(result, sql, parameters);
190    
191                if (result.rs.next()) {
192    
193                    if (readPrimitive) {
194                        return (T) readPrimitive(result.rs, 1, objectClass);
195    
196                    } else {
197                        T t = objectClass.newInstance();
198                        readObject(t, result.rs);
199                        return t;
200                    }
201                }
202    
203                return null;
204    
205            } catch (IllegalAccessException e) {
206                throw new PersismException(e);
207    
208            } catch (InvocationTargetException e) {
209                throw new PersismException(e);
210    
211            } catch (SQLException e) {
212                throw new PersismException(e);
213    
214            } catch (InstantiationException e) {
215                throw new PersismException(e);
216    
217            } catch (IOException e) {
218                throw new PersismException(e);
219    
220            } finally {
221                Util.cleanup(result.st, result.rs);
222            }
223        }

225        private <T> T readObject(Object object, ResultSet rs) throws IllegalAccessException, SQLException, InvocationTargetException, IOException {
226    
227            Class objectClass = object.getClass();
228            // We should never call this method with a primitive type.
229            assert Types.getType(objectClass) == null;
230    
231            Map<String, PropertyInfo> properties;
232            if (objectClass.getAnnotation(QueryResult.class) == null) {
233                properties = metaData.getTableColumns(objectClass, connection);
234            } else {
235                properties = metaData.getQueryColumns(objectClass, rs);
236            }
237    
238            // Test if all properties have column mapping and throw PersismException if not
239            // This block verifies that the object is fully initialized.
240            // Any properties not marked by NotMapped should have been set (or if they have a getter only)
241            // If not throw a PersismException
242            Collection<PropertyInfo> allProperties = MetaData.getPropertyInfo(objectClass);
243            if (properties.values().size() < allProperties.size()) {
244                Set<PropertyInfo> missing = new HashSet<PropertyInfo>(allProperties.size());
245                missing.addAll(allProperties);
246                missing.removeAll(properties.values());
247    
248                StringBuilder sb = new StringBuilder();
249                String sep = "";
250                for (PropertyInfo prop : missing) {
251                    sb.append(sep).append(prop.propertyName);
252                    sep = ",";
253                }
254    
255                throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not found in the queried columns (" + sb + ").");
256            }
257    
258    
259            ResultSetMetaData rsmd = rs.getMetaData();
260            int columnCount = rsmd.getColumnCount();
261            List<String> foundColumns = new ArrayList<String>(columnCount);
262    
263            for (int j = 1; j <= columnCount; j++) {
264    
265                String columnName = rsmd.getColumnLabel(j);
266                PropertyInfo columnProperty = properties.get(columnName);
267    
268                if (columnProperty != null) {
269                    Class getterType = columnProperty.getter.getReturnType();
270    
271                    Object value = readPrimitive(rs, j, getterType);
272    
273                    foundColumns.add(columnName);
274    
275                    if (value != null) {
276                        try {
277                            columnProperty.setter.invoke(object, value);
278                        } catch (IllegalArgumentException e) {
279                            String msg = e.getMessage() + " Object " + objectClass + ". Column: " + columnName + " Type of property: " + getterType + " - Type read: " + value.getClass() + " VALUE: " + value;
280                            throw new PersismException(msg, e);
281                        }
282    
283                    }
284                }
285            }
286    
287            if (foundColumns.size() < properties.keySet().size()) {
288    
289                Set<String> missing = new HashSet<String>(columnCount);
290                missing.addAll(properties.keySet());
291                missing.removeAll(foundColumns);
292    
293                throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not found in the queried columns. : " + missing);
294            }
295    
296            if (object instanceof Persistable) {
297                // Save this object's initial state to later detect changed properties
298                ((Persistable) object).saveReadState();
299            }
300    
301            return (T) object;
302    
303        }

305        private Object readPrimitive(ResultSet rs, int column, Class returnType) throws SQLException, IOException {
306            ResultSetMetaData resultSetMetaData = rs.getMetaData();
307            int sqlColumnType = resultSetMetaData.getColumnType(column);
308    
309            Types columnType = Types.convert(sqlColumnType); // note this could be null if we can't match a type
310            // Since there is no specific SQL column type for UUID we will use the return type to detect it.
311            if (returnType.equals(UUID.class)) {
312                // Check the return type for UUID since resultSetMetaData.getColumnType(column) has no UUID type
313                // it always returns a char or nvarchar so we'll just test and set it here. FFS.
314                columnType = Types.UUIDType;
315            }
316            String columnName = resultSetMetaData.getColumnLabel(column);
317    
318            Object value;
319            if (columnType != null) {
320                switch (columnType) {
321    
322                    case TimestampType:
323                        // work around to Oracle reading a oracle.sql.TIMESTAMP class with getObject
324                        value = rs.getTimestamp(column);
325                        break;
326                    case ClobType:
327                        value = rs.getClob(column);
328                        InputStream in = ((Clob) value).getAsciiStream();
329                        StringWriter write = new StringWriter();
330    
331                        int c = -1;
332                        while ((c = in.read()) != -1) {
333                            write.write(c);
334                        }
335                        write.flush();
336                        value = write.toString();
337                        break;
338                    case BlobType:
339                        // todo BlobType
340                    case InputStreamType:
341                        // todo InputStreamType
342                    case ReaderType:
343                        // todo ReaderType
344                    case EnumType:
345                        // todo EnumType?
346                    case UUIDType:
347                        value = rs.getObject(column);
348                        if (value != null) {
349                            value = UUID.fromString("" + value);
350                        }
351                        break;
352                    case floatType:
353                    case FloatType:
354                    case doubleType:
355                    case DoubleType:
356                        if (returnType.equals(java.math.BigDecimal.class)) {
357                            value = rs.getBigDecimal(column);
358                        } else if (returnType.equals(java.lang.Double.class)) {
359                            value = rs.getDouble(column);
360                        } else {
361                            value = rs.getFloat(column);
362                        }
363                        break;
364    
365                    default:
366                        value = rs.getObject(column);
367                }
368    
369            } else {
370                log.warn("Column type not known for SQL type " + sqlColumnType);
371                value = rs.getObject(column);
372            }
373    
374            // If value is null or column type is unknown - no need to try to convert anything.
375            if (value != null && columnType != null) {
376    
377                Types valueType = Types.getType(value.getClass());
378    
379                // try to convert or cast the value to the proper type.
380                // todo do code coverage for each specific type
381                switch (valueType) {
382    
383                    case booleanType:
384                    case BooleanType:
385                        break;
386    
387                    case byteType:
388                    case ByteType:
389                    case shortType:
390                    case ShortType:
391                    case integerType:
392                    case IntegerType:
393                        // int to bool
394                        if (returnType == Boolean.class || returnType == boolean.class) {
395                            value = (Integer.valueOf("" + value) == 0) ? false : true;
396                        }
397                        break;
398    
399                    case longType:
400                    case LongType:
401                        // long to date
402                        if (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
403                            long lval = Long.valueOf("" + value);
404    
405                            if (returnType.equals(java.sql.Date.class)) {
406                                value = new java.sql.Date(lval);
407                            } else {
408                                value = new java.util.Date(lval);
409                            }
410                        } else if (returnType == Integer.class || returnType == int.class) {
411                            log.warn("Possible overflow column " + columnName + " - Property is INT and column value is LONG");
412                            value = Integer.parseInt("" + value);
413                        }
414    
415                        break;
416    
417                    case floatType:
418                    case FloatType:
419                        break;
420    
421                    case doubleType:
422                    case DoubleType:
423                        // float or doubles to BigDecimal
424                        if (returnType == BigDecimal.class) {
425                            value = new BigDecimal("" + value);
426                        } else if (returnType == Float.class || returnType == float.class) {
427                            // todo add tests for this
428                            log.warn("Possible overflow column " + columnName + " - Property is FLOAT and column value is DOUBLE");
429                            value = Float.parseFloat("" + value);
430                        } else if (returnType == Integer.class || returnType == int.class) {
431                            log.warn("Possible overflow column " + columnName + " - Property is INT and column value is DOUBLE");
432                            String val = "" + value;
433                            if (val.contains(".")) {
434                                val = val.substring(0, val.indexOf("."));
435                            }
436                            value = Integer.parseInt(val);
437                        }
438                        break;
439    
440                    case BigDecimalType:
441                        // mostly oracle
442                        if (returnType == Float.class || returnType == float.class) {
443                            value = ((BigDecimal) value).floatValue();
444                        } else if (returnType == Double.class || returnType == double.class) {
445                            value = ((BigDecimal) value).doubleValue();
446                        } else if (returnType == Long.class || returnType == long.class) {
447                            value = ((BigDecimal) value).longValue();
448                        } else if (returnType == Integer.class || returnType == int.class) {
449                            value = ((BigDecimal) value).intValue();
450                        } else if (returnType == Boolean.class || returnType == boolean.class) {
451                            value = ((BigDecimal) value).intValue() == 1;
452                        }
453                        break;
454    
455                    case StringType:
456    
457                        // Read a string but we want a date
458                        if (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
459                            // This condition occurs in SQLite when you have a datetime with default annotated
460                            // the format returned is 2012-06-02 19:59:49
461                            Date dval = null;
462                            try {
463                                // Used for SQLite returning dates as Strings under some conditions
464                                DateFormat df = new SimpleDateFormat("yyyy-MM-DD hh:mm:ss");
465                                dval = df.parse("" + value);
466                            } catch (ParseException e) {
467                                String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + returnType + " - Type read: " + value.getClass() + " VALUE: " + value;
468                                throw new PersismException(msg, e);
469                            }
470    
471                            if (returnType.equals(java.sql.Date.class)) {
472                                value = new java.sql.Date(dval.getTime());
473                            } else {
474                                value = dval;
475                            }
476    
477                        } else if (returnType.isEnum()) {
478                            // If this is an enum do a case insensitive comparison
479                            Object[] enumConstants = returnType.getEnumConstants();
480                            for (Object element : enumConstants) {
481                                if (("" + value).equalsIgnoreCase(element.toString())) {
482                                    value = element;
483                                    break;
484                                }
485                            }
486                        }
487    
488                        break;
489    
490                    case characterType:
491                    case CharacterType:
492                        break;
493    
494                    case UtilDateType:
495                        break;
496                    case SQLDateType:
497                        break;
498                    case TimeType:
499                        break;
500    
501                    case TimestampType:
502                        if (returnType.isAssignableFrom(Date.class) || returnType.isAssignableFrom(java.sql.Date.class)) {
503                            if (returnType.equals(java.sql.Date.class)) {
504                                value = new java.sql.Date(((Timestamp) value).getTime());
505                            } else {
506                                value = new Date(((Timestamp) value).getTime());
507                            }
508                        } else {
509                            value = ((Timestamp) value).getTime();
510                        }
511    
512                        break;
513    
514                    case byteArrayType:
515                        break;
516                    case ByteArrayType:
517                        break;
518                    case charArrayType:
519                        break;
520                    case CharArrayType:
521                        break;
522                    case ClobType:
523                        // Convert to string
524                        if (value != null) {
525                            value = "" + value;
526                        }
527                        break;
528                    case BlobType:
529                        break;
530                    case InputStreamType:
531                        break;
532                    case ReaderType:
533                        break;
534                    case EnumType:
535                        break;
536                }
537            }
538    
539            return value;
540    
541    /*
542                } else if (type == java.io.InputStream.class) {
543                    value = resultSet.getBinaryStream(column);
544                } else if (type == java.io.Reader.class) {
545                    value = resultSet.getCharacterStream(column);
546                } else if (type == java.sql.Clob.class) {
547                    value = resultSet.getClob(column);
548                } else if (type == java.sql.Blob.class) {
549                    value = resultSet.getBlob(column);

554        private Object XreadPrimitive(ResultSet rs, int column, Class returnType) throws SQLException {
555            // todo in H2
556            // CREATE TABLE X  ID INT IDENTITY PRIMARY KEY -- creates INT
557            // CREATE TABLE X  ID IDENTITY PRIMARY KEY -- creates LONG - can cause problems if data object uses int
558    
559            ResultSetMetaData resultSetMetaData = rs.getMetaData();
560            Types columnType = Types.convert(resultSetMetaData.getColumnType(column)); // note this could be null if we can't match a type
561            String columnName = resultSetMetaData.getColumnLabel(column);
562    
563            Object value;
564            if (Types.TimestampType == columnType) {
565                // work around to Oracle reading a oracle.sql.TIMESTAMP class with getObject
566                value = rs.getTimestamp(column);
567            } else {
568                value = rs.getObject(column);
569            }
570    
571            if (value != null) {
572    
573                Types valueType = Types.getType(value.getClass());
574    
575                //if (log.isDebugEnabled()) {
576                log.debug("COLUMN: " + columnName + " value " + value + " type: " + valueType + " COLUMN TYPE: " + columnType + " return type: " + returnType);
577    //            }
578    
579                // Convert value to native java type
580                if (value instanceof Timestamp) {
581    // done
582                    if (!returnType.isAssignableFrom(java.util.Date.class) && !returnType.isAssignableFrom(java.sql.Date.class)) {
583                        value = ((Timestamp) value).getTime();
584                    } else {
585                        if (returnType.equals(java.sql.Date.class)) {
586                            value = new java.sql.Date(((Timestamp) value).getTime());
587                        } else {
588                            value = new java.util.Date(((Timestamp) value).getTime());
589                        }
590                    }
591    
592                } else if (value instanceof Long &&
593                        (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class))) {
594    // done
595    
596                    long lval = Long.valueOf("" + value);
597    
598                    if (returnType.equals(java.sql.Date.class)) {
599                        value = new java.sql.Date(lval);
600                    } else {
601                        value = new java.util.Date(lval);
602                    }
603                } else if (value instanceof String &&
604                        (returnType.isAssignableFrom(java.util.Date.class) || returnType.isAssignableFrom(java.sql.Date.class))) {
605    // done
606                    // This condition occurs in SQLite when you have a datetime with default annotated
607                    // the format returned is 2012-06-02 19:59:49
608                    Date dval = null;
609                    try {
610                        // Used for SQLite returning dates as Strings under some conditions
611                        DateFormat df = new SimpleDateFormat("yyyy-MM-DD hh:mm:ss");
612                        dval = df.parse("" + value);
613                    } catch (ParseException e) {
614                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + returnType + " - Type read: " + value.getClass() + " VALUE: " + value;
615                        throw new PersismException(msg, e);
616                    }
617    
618                    if (returnType.equals(java.sql.Date.class)) {
619                        value = new java.sql.Date(dval.getTime());
620                    } else {
621                        value = dval;
622                    }
623    
624                } else if (value instanceof String && returnType.isEnum()) {
625    // done
626                    // If this is an enum do a case insensitive comparison
627                    Object[] enumConstants = returnType.getEnumConstants();
628                    for (Object element : enumConstants) {
629                        if (("" + value).equalsIgnoreCase(element.toString())) {
630                            value = element;
631                            break;
632                        }
633                    }
634                } else if (value instanceof BigDecimal) { //  mostly oracle.
635    // done
636                    if (returnType == Float.class || returnType == float.class) {
637                        value = ((BigDecimal) value).floatValue();
638                    } else if (returnType == Double.class || returnType == double.class) {
639                        value = ((BigDecimal) value).doubleValue();
640                    } else if (returnType == Long.class || returnType == long.class) {
641                        value = ((BigDecimal) value).longValue();
642                    } else if (returnType == Integer.class || returnType == int.class) {
643                        value = ((BigDecimal) value).intValue();
644                    } else if (returnType == Boolean.class || returnType == boolean.class) {
645                        value = ((BigDecimal) value).intValue() == 1;
646                    }
647    
648                } else if (value instanceof Integer && (returnType == Boolean.class || returnType == boolean.class)) {
649    // done
650                    value = (Integer.valueOf("" + value) == 0) ? false : true;
651    
652                } else if (value instanceof Float && returnType == BigDecimal.class) {
653    // done
654                    value = new BigDecimal("" + value);
655                }
656            }
657    
658            return value;
659        }

662        private Result exec(Result result, String sql, Object... parameters) throws SQLException {
663            if (sql.toLowerCase().contains("select ")) {
664                result.st = connection.prepareStatement(sql);
665    
666                PreparedStatement pst = (PreparedStatement) result.st;
667                Util.setParameters(pst, parameters);
668                result.rs = pst.executeQuery();
669            } else {
670                // todo unit tests need to cover this.
671                result.st = connection.prepareCall("{call " + sql + "}");
672    
673                CallableStatement cst = (CallableStatement) result.st;
674                Util.setParameters(cst, parameters);
675                result.rs = cst.executeQuery();
676            }
677            return result;
678        }
