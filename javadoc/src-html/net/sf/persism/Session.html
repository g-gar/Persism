067    /**
068     * Updates the data object in the database.
069     *
070     * @param object data object to update.
071     * @return usually 1 to indicate rows changed via JDBC.
072     * @throws PersismException Indicating the upcoming robot uprising.
073     */
074    public int update(Object object) throws PersismException {
075        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
076        if (primaryKeys.size() == 0) {
077            throw new PersismException("Cannot perform UPDATE - " + metaData.getTableName(object.getClass()) + " has no primary keys.");
078        }
079
080        PreparedStatement st = null;
081        try {
082
083            String updateStatement = null;
084            try {
085                updateStatement = metaData.getUpdateStatement(object, connection);
086            } catch (NoChangesDetectedForUpdateException e) {
087                log.info("No properties changed. No update required for Object: " + object + " class: " + object.getClass().getName());
088                return 0;
089            }
090
091            st = connection.prepareStatement(updateStatement);
092
093            // These keys should always be in sorted order.
094            Map<String, PropertyInfo> allProperties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
095            Map<String, PropertyInfo> changedProperties;
096            if (object instanceof Persistable) {
097                changedProperties = metaData.getChangedProperties((Persistable) object, connection);
098            } else {
099                changedProperties = allProperties;
100            }
101
102            List<Object> params = new ArrayList<>(primaryKeys.size());
103            List<ColumnInfo> columnInfos = new ArrayList<>(primaryKeys.size());
104
105            Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
106
107            for (String column : changedProperties.keySet()) {
108                ColumnInfo columnInfo = columns.get(column);
109
110                if (!primaryKeys.contains(column)) {
111                    Object value = allProperties.get(column).getter.invoke(object);
112
113                    if (value instanceof String) {
114                        // check width
115                        String str = (String) value;
116                        if (str.length() > columnInfo.length) {
117                            str = str.substring(0, columnInfo.length);
118                            log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
119                            value = str;
120                        }
121                    }
122                    params.add(value);
123                    columnInfos.add(columnInfo);
124                }
125            }
126
127            for (String column : primaryKeys) {
128                params.add(allProperties.get(column).getter.invoke(object));
129                columnInfos.add(metaData.getColumns(object.getClass(), connection).get(column));
130            }
131            assert params.size() == columnInfos.size();
132            for (int j = 0; j < params.size(); j++) {
133                if (params.get(j) != null) {
134                    params.set(j, convert(params.get(j), columnInfos.get(j).columnType.getJavaType(), columnInfos.get(j).columnName));
135                }
136            }
137            setParameters(st, params.toArray());
138            int ret = st.executeUpdate();
139
140            if (object instanceof Persistable) {
141                // Save this object state to later detect changed properties
142                ((Persistable) object).saveReadState();
143            }
144
145            return ret;
146
147        } catch (Exception e) {
148            Util.rollback(connection);
149            throw new PersismException(e.getMessage(), e);
150
151        } finally {
152            Util.cleanup(st, null);
153        }
154    }

156    /**
157     * Inserts the data object in the database refreshing with autoinc and other defaults that may exist.
158     *
159     * @param object the data object to insert.
160     * @return usually 1 to indicate rows changed via JDBC.
161     * @throws PersismException When planet of the apes starts happening.
162     */
163    public int insert(Object object) throws PersismException {
164        String insertStatement = metaData.getInsertStatement(object, connection);
165
166        PreparedStatement st = null;
167        ResultSet rs = null;
168
169        try {
170            // These keys should always be in sorted order.
171            Map<String, PropertyInfo> properties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
172            Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
173
174            List<String> generatedKeys = new ArrayList<>(1);
175            for (ColumnInfo column : columns.values()) {
176                if (column.autoIncrement) {
177                    generatedKeys.add(column.columnName);
178                } else if (metaData.connectionType == ConnectionTypes.PostgreSQL && column.primary && column.hasDefault) {
179                    generatedKeys.add(column.columnName);
180                }
181            }
182
183            if (generatedKeys.size() > 0) {
184                String[] keyArray = generatedKeys.toArray(new String[0]);
185                st = connection.prepareStatement(insertStatement, keyArray);
186            } else {
187                st = connection.prepareStatement(insertStatement);
188            }
189
190            boolean tableHasDefaultColumnValues = false;
191
192            List<Object> params = new ArrayList<>(columns.size());
193            List<ColumnInfo> columnInfos = new ArrayList<>(columns.size());
194
195            for (ColumnInfo columnInfo : columns.values()) {
196
197                PropertyInfo propertyInfo = properties.get(columnInfo.columnName);
198                if (!columnInfo.autoIncrement) {
199
200                    if (columnInfo.hasDefault) {
201                        // Do not include if this column has a default and no value has been
202                        // set on it's associated property.
203
204                        if (propertyInfo.getter.getReturnType().isPrimitive()) {
205                            warnNoDuplicates("Property " + propertyInfo.propertyName + " for column " + columnInfo.columnName + " for class " + object.getClass() +
206                                    " should be an Object type to properly detect NULL for defaults (change it from the primitive type to its Boxed version).");
207                        }
208
209                        if (propertyInfo.getter.invoke(object) == null) {
210
211                            if (columnInfo.primary) {
212                                // This is supported with PostgreSQL but otherwise throw this an exception
213                                if (!(metaData.connectionType == ConnectionTypes.PostgreSQL)) {
214                                    throw new PersismException("Non-auto inc generated primary keys are not supported. Please assign your primary key value before performing an insert.");
215                                }
216                            }
217
218                            tableHasDefaultColumnValues = true;
219                            continue;
220                        }
221                    }
222
223                    Object value = propertyInfo.getter.invoke(object);
224
225                    if (value instanceof String) {
226                        // check width
227                        String str = (String) value;
228                        if (str.length() > columnInfo.length) {
229                            str = str.substring(0, columnInfo.length);
230                            log.warn("TRUNCATION with Column: " + columnInfo.columnName + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
231                            value = str;
232                        }
233                    }
234                    params.add(value);
235                    columnInfos.add(columnInfo);
236                }
237            }
238
239            // https://forums.oracle.com/forums/thread.jspa?threadID=879222
240            // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
241            //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
242            assert params.size() == columnInfos.size();
243            for (int j = 0; j < params.size(); j++) {
244                if (params.get(j) != null) {
245                    params.set(j, convert(params.get(j), columnInfos.get(j).columnType.getJavaType(), columnInfos.get(j).columnName));
246                }
247            }
248
249            setParameters(st, params.toArray());
250            int ret = st.executeUpdate();
251            if (log.isDebugEnabled()) {
252                log.debug("insert ret: " + ret);
253            }
254
255            if (generatedKeys.size() > 0) {
256                rs = st.getGeneratedKeys();
257                for (String column : generatedKeys) {
258                    if (rs.next()) {
259
260                        Method setter = properties.get(column).setter;
261                        Object value = getTypedValueReturnedFromGeneratedKeys(setter.getParameterTypes()[0], rs);
262
263                        if (log.isDebugEnabled()) {
264                            log.debug(column + " generated " + value);
265                        }
266                        setter.invoke(object, value);
267                    }
268                }
269            }
270
271            if (tableHasDefaultColumnValues) {
272                // Read the full object back to update any properties which had defaults
273                fetch(object);
274            } else {
275                if (object instanceof Persistable) {
276                    // Save this object new state to later detect changed properties
277                    ((Persistable) object).saveReadState();
278                }
279            }
280
281            return ret;
282        } catch (Exception e) {
283            Util.rollback(connection);
284            throw new PersismException(e.getMessage(), e);
285        } finally {
286            Util.cleanup(st, rs);
287        }
288    }

298    public int delete(Object object) throws PersismException {
299
300        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
301        if (primaryKeys.size() == 0) {
302            throw new PersismException("Cannot perform DELETE - " + metaData.getTableName(object.getClass()) + " has no primary keys.");
303        }
304
305        PreparedStatement st = null;
306        try {
307            String deleteStatement = metaData.getDeleteStatement(object, connection);
308            st = connection.prepareStatement(deleteStatement);
309
310            // These keys should always be in sorted order.
311            Map<String, PropertyInfo> columns = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
312
313            List<Object> params = new ArrayList<>(primaryKeys.size());
314            List<ColumnInfo> columnInfos = new ArrayList<>(primaryKeys.size());
315            for (String column : primaryKeys) {
316                params.add(columns.get(column).getter.invoke(object));
317                columnInfos.add(metaData.getColumns(object.getClass(), connection).get(column));
318            }
319
320            for (int j = 0; j < params.size(); j++) {
321                if (params.get(j) != null) {
322                    params.set(j, convert(params.get(j), columnInfos.get(j).columnType.getJavaType(), columnInfos.get(j).columnName));
323                }
324            }
325            setParameters(st, params.toArray());
326            int ret = st.executeUpdate();
327            return ret;
328
329        } catch (Exception e) {
330            Util.rollback(connection);
331            throw new PersismException(e.getMessage(), e);
332
333        } finally {
334            Util.cleanup(st, null);
335        }

365    /**
366     * Query for a list of objects of the specified class using the specified SQL query and parameters.
367     * The type of the list can be Data Objects or native Java Objects or primitives.
368     *
369     * @param objectClass class of objects to return.
370     * @param sql         query string to execute.
371     * @param parameters  parameters to the query.
372     * @param <T>         Return type
373     * @return a list of objects of the specified class using the specified SQL query and parameters.
374     * @throws PersismException If something goes wrong you get a big stack trace.
375     */
376    public <T> List<T> query(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
377        List<T> list = new ArrayList<T>(32);
378
379        Result result = new Result();
380
381        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
382        boolean isPOJO = Types.getType(objectClass) == null;
383
384        if (isPOJO && objectClass.getAnnotation(NotTable.class) == null) {
385            // Make sure columns are initialized if this is a table.
386            metaData.getTableColumnsPropertyInfo(objectClass, connection);
387        }
388
389        try {
390
391            exec(result, sql, parameters);
392
393            while (result.rs.next()) {
394                if (isPOJO) {
395                    // should be getDeclaredConstructor().newInstance() now.
396                    T t = objectClass.newInstance();
397                    list.add(readObject(t, result.rs));
398                } else {
399                    list.add((T) readColumn(result.rs, 1, objectClass));
400                }
401            }
402
403        } catch (Exception e) {
404            Util.rollback(connection);
405            throw new PersismException(e.getMessage(), e);
406        } finally {
407            Util.cleanup(result.st, result.rs);
408        }
409
410        return list;
411
412    }

414    /**
415     * Fetch an object from the database by it's primary key(s).
416     * You should instantiate the object and set the primary key properties before calling this method.
417     *
418     * @param object Data object to read from the database.
419     * @return true if the object was found by the primary key.
420     * @throws PersismException if something goes wrong.
421     */
422    public boolean fetch(Object object) throws PersismException {
423        Class<?> objectClass = object.getClass();
424
425        // If we know this type it means it's a primitive type. This method cannot be used for primitives
426        boolean readPrimitive = Types.getType(objectClass) != null;
427        if (readPrimitive) {
428            // For unit tests
429            throw new PersismException("Cannot read a primitive type object with this method.");
430        }
431
432        List<String> primaryKeys = metaData.getPrimaryKeys(objectClass, connection);
433        if (primaryKeys.size() == 0) {
434            throw new PersismException("Cannot perform FETCH - " + metaData.getTableName(objectClass) + " has no primary keys.");
435        }
436
437        Map<String, PropertyInfo> properties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
438        List<Object> params = new ArrayList<>(primaryKeys.size());
439        List<ColumnInfo> columnInfos = new ArrayList<>(primaryKeys.size());
440        Map<String, ColumnInfo> cols = metaData.getColumns(objectClass, connection);
441        Result result = new Result();
442        try {
443            for (String column : primaryKeys) {
444                PropertyInfo propertyInfo = properties.get(column);
445                params.add(propertyInfo.getter.invoke(object));
446                columnInfos.add(cols.get(column));
447            }
448            assert params.size() == columnInfos.size();
449
450            String sql = metaData.getSelectStatement(object, connection);
451            if (log.isDebugEnabled()) {
452                log.debug("FETCH " + sql + " PARAMS: " + params);
453            }
454            for (int j = 0; j < params.size(); j++) {
455                if (params.get(j) != null) {
456                    params.set(j, convert(params.get(j), columnInfos.get(j).columnType.getJavaType(), columnInfos.get(j).columnName));
457                }
458            }
459            exec(result, sql, params.toArray());
460
461            if (result.rs.next()) {
462                readObject(object, result.rs);
463                return true;
464            }
465            return false;
466
467        } catch (Exception e) {
468            Util.rollback(connection);
469            throw new PersismException(e.getMessage(), e);
470        } finally {
471            Util.cleanup(result.st, result.rs);
472        }
473    }

496    /**
497     * Fetch an object of the specified type from the database. The type can be a Data Object or a native Java Object or primitive.
498     *
499     * @param objectClass Type of returned value
500     * @param sql         query - this would usually be a select OR a select of a single column if the type is a primitive.
501     *                    If this is a primitive type then this method will only look at the 1st column in the result.
502     * @param parameters  parameters to the query.
503     * @param <T>         Return type
504     * @return value read from the database of type T or null if not found
505     * @throws PersismException Well, this is a runtime exception so actually it could be anything really.
506     */
507    public <T> T fetch(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
508        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
509        boolean isPOJO = Types.getType(objectClass) == null;
510
511        Result result = new Result();
512        try {
513
514            exec(result, sql, parameters);
515
516            if (result.rs.next()) {
517
518                if (isPOJO) {
519                    T t = objectClass.newInstance();
520                    readObject(t, result.rs);
521                    return t;
522                } else {
523                    return (T) readColumn(result.rs, 1, objectClass);
524                }
525            }
526
527            return null;
528
529        } catch (Exception e) {
530            Util.rollback(connection);
531            throw new PersismException(e.getMessage(), e);
532        } finally {
533            Util.cleanup(result.st, result.rs);
534        }
535    }

545    private Result exec(Result result, String sql, Object... parameters) throws SQLException {
546        if (sql.toLowerCase().startsWith("select ")) {
547            result.st = connection.prepareStatement(sql);
548
549            PreparedStatement pst = (PreparedStatement) result.st;
550            setParameters(pst, parameters);
551            result.rs = pst.executeQuery();
552        } else {
553            if (!sql.toLowerCase().startsWith("{call")) {
554                sql = "{call " + sql + "} ";
555            }
556            result.st = connection.prepareCall(sql);
557
558            CallableStatement cst = (CallableStatement) result.st;
559            setParameters(cst, parameters);
560            result.rs = cst.executeQuery();
561        }
562        return result;
563    }

565    private <T> T readObject(Object object, ResultSet rs) throws IllegalAccessException, SQLException, InvocationTargetException, IOException {
566
567        Class<?> objectClass = object.getClass();
568        // We should never call this method with a primitive type.
569        assert Types.getType(objectClass) == null;
570
571        Map<String, PropertyInfo> properties;
572        if (objectClass.getAnnotation(NotTable.class) == null) {
573            properties = metaData.getTableColumnsPropertyInfo(objectClass, connection);
574        } else {
575            properties = metaData.getQueryColumnsPropertyInfo(objectClass, rs);
576        }
577
578        // Test if all properties have column mapping and throw PersismException if not
579        // This block verifies that the object is fully initialized.
580        // Any properties not marked by NotMapped should have been set (or if they have a getter only)
581        // If not throw a PersismException
582        Collection<PropertyInfo> allProperties = MetaData.getPropertyInfo(objectClass);
583        if (properties.values().size() < allProperties.size()) {
584            Set<PropertyInfo> missing = new HashSet<>(allProperties.size());
585            missing.addAll(allProperties);
586            missing.removeAll(properties.values());
587
588            StringBuilder sb = new StringBuilder();
589            String sep = "";
590            for (PropertyInfo prop : missing) {
591                sb.append(sep).append(prop.propertyName);
592                sep = ",";
593            }
594
595            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not initialized in the queried columns (" + sb + ").");
596        }
597
598        ResultSetMetaData rsmd = rs.getMetaData();
599        int columnCount = rsmd.getColumnCount();
600        List<String> foundColumns = new ArrayList<>(columnCount);
601
602        for (int j = 1; j <= columnCount; j++) {
603
604            String columnName = rsmd.getColumnLabel(j);
605            PropertyInfo columnProperty = properties.get(columnName);
606
607            if (columnProperty != null) {
608                Class<?> returnType = columnProperty.getter.getReturnType();
609
610                Object value = readColumn(rs, j, returnType);
611
612                foundColumns.add(columnName);
613
614                if (value != null) {
615                    try {
616                        columnProperty.setter.invoke(object, value);
617                    } catch (IllegalArgumentException e) {
618                        String msg = "Object " + objectClass + ". Column: " + columnName + " Type of property: " + returnType + " - Type read: " + value.getClass() + " VALUE: " + value;
619                        throw new PersismException(msg, e);
620                    }
621                }
622            }
623        }
624
625        // This is doing a similar check to above but on the ResultSet itself.
626        // This tests for when a user writes their own SQL and forgets a column.
627        if (foundColumns.size() < properties.keySet().size()) {
628
629            Set<String> missing = new HashSet<>(columnCount);
630            missing.addAll(properties.keySet());
631            missing.removeAll(foundColumns);
632
633            // todo maybe strict mode off logs warn? Should we do this if this is Query vs Table?
634            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not initialized by the queried columns: " + foundColumns + " Missing:" + missing);
635        }
636
637        if (object instanceof Persistable) {
638            // Save this object initial state to later detect changed properties
639            ((Persistable) object).saveReadState();
640        }
641
642        return (T) object;
643    }

645    private Object readColumn(ResultSet rs, int column, Class<?> returnType) throws SQLException, IOException {
646        ResultSetMetaData rsmd = rs.getMetaData();
647        int sqlColumnType = rsmd.getColumnType(column);
648        String columnName = rsmd.getColumnLabel(column);
649
650        if (returnType.isEnum()) {
651            // Some DBs may read an enum type as other 1111 - we can tell it here to read it as a string.
652            sqlColumnType = java.sql.Types.CHAR;
653        }
654
655        Object value;
656
657        Types columnType = Types.convert(sqlColumnType); // note this could be null if we can't match a type
658        if (columnType != null) {
659
660            switch (columnType) {
661
662                case TimestampType:
663                    if (returnType.equals(String.class)) { // JTDS
664                        value = rs.getString(column);
665                    } else {
666                        // work around to Oracle reading a oracle.sql.TIMESTAMP class with getObject
667                        value = rs.getTimestamp(column);
668                    }
669                    break;
670
671                case ByteArrayType:
672                case byteArrayType:
673                    value = rs.getBytes(column);
674                    break;
675
676                case ClobType:
677                    Clob clob = rs.getClob(column);
678                    try (InputStream in = clob.getAsciiStream()) {
679                        StringWriter writer = new StringWriter();
680
681                        int c = -1;
682                        while ((c = in.read()) != -1) {
683                            writer.write(c);
684                        }
685                        writer.flush();
686                        value = writer.toString();
687                    }
688                    break;
689
690                case BlobType:
691                    byte[] buffer = new byte[1024];
692                    Blob blob = rs.getBlob(column);
693                    try (InputStream in = blob.getBinaryStream()) {
694                        ByteArrayOutputStream bos = new ByteArrayOutputStream((int) blob.length());
695                        for (int len; (len = in.read(buffer)) != -1; ) {
696                            bos.write(buffer, 0, len);
697                        }
698                        value = bos.toByteArray();
699                    }
700                    break;
701
702                case IntegerType:
703                    // stupid SQLite reports LONGS as Integers for date types which WRAPS past Integer.MAX - Clowns.
704                    if (metaData.connectionType == ConnectionTypes.SQLite) {
705                        value = rs.getObject(column);
706                        if (value != null) {
707                            if (value instanceof Long) {
708                                value = rs.getLong(column);
709                            } else {
710                                value = rs.getInt(column);
711                            }
712                        }
713                    } else {
714                        value = rs.getObject(column) == null ? null : rs.getInt(column);
715                    }
716                    break;
717
718                case LongType:
719                    value = rs.getObject(column) == null ? null : rs.getLong(column);
720                    break;
721
722                case FloatType:
723                    value = rs.getObject(column) == null ? null : rs.getFloat(column);
724                    break;
725
726                case DoubleType:
727                    value = rs.getObject(column) == null ? null : rs.getDouble(column);
728                    break;
729
730                case BigIntegerType:
731                case BigDecimalType:
732                    value = null;
733                    if (returnType.equals(BigInteger.class)) {
734                        BigDecimal bd = rs.getBigDecimal(column);
735                        if (bd != null) {
736                            value = bd.toBigInteger();
737                        }
738                    } else {
739                        value = rs.getBigDecimal(column);
740                    }
741                    break;
742
743                case TimeType:
744                    value = rs.getTime(column);
745                    break;
746
747// We can't assume rs.getDate will work. SQLITE actually has a long value in here.
748// We can live with rs.getObject and the convert method will handle it.
749//                case SQLDateType:
750//                case UtilDateType:
751//                    value = rs.getDate(column);
752//                    break;
753
754                case StringType:
755                    value = rs.getString(column);
756                    break;
757
758                default:
759                    value = rs.getObject(column);
760            }
761
762        } else {
763            log.warn("Column type not known for SQL type " + sqlColumnType, new Throwable());
764            value = rs.getObject(column);
765        }
766
767        // If value is null or column type is unknown - no need to try to convert anything.
768        if (value != null && columnType != null) {
769            value = convert(value, returnType, columnName);
770        }
771
772        return value;
773    }

775    // Make a sensible conversion of the value type from the DB and the property type defined
776    // on the Data class - or the value type from the property to the statement parameter.
777    private Object convert(Object value, Class<?> targetType, String columnName) {
778        assert value != null;
779
780        Types valueType = Types.getType(value.getClass());
781
782        if (valueType == null) {
783            log.warn("Conversion: Unknown Persism type " + value.getClass() + " - no conversion performed.");
784            return value;
785        }
786
787        Object returnValue = value;
788
789        // try to convert or cast the value to the proper type.
790        switch (valueType) {
791
792            case booleanType:
793            case BooleanType:
794                log.debug("BooleanType");
795                break;
796
797            case byteType:
798            case ByteType:
799            case shortType:
800            case ShortType:
801                log.debug(valueType);
802                break;
803            // TODO test out short and byte - JDBC tends to read single byte as short
804            // Might need byte to bool also upcasting can cause errors
805            case integerType:
806            case IntegerType:
807                // int to bool
808                if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
809                    returnValue = Integer.valueOf("" + value) == 0 ? false : true;
810
811                } else if (targetType.equals(Time.class)) {
812                    // SQLite when a Time is defined VIA a convert from LocalTime via Time.valueOf (see getContactForTest)
813                    returnValue = new Time(((Integer) value).longValue());
814
815                } else if (targetType.equals(LocalTime.class)) {
816                    // SQLite for Time SQLite sees Long, for LocalTime it sees Integer
817                    returnValue = new Time((Integer) value).toLocalTime();
818                }
819                break;
820
821            case longType:
822            case LongType:
823                long lval = Long.valueOf("" + value);
824                if (targetType.equals(java.sql.Date.class)) {
825                    returnValue = new java.sql.Date(lval);
826
827                } else if (targetType.equals(java.util.Date.class)) {
828                    returnValue = new java.util.Date(lval);
829
830                } else if (targetType.equals(Timestamp.class)) {
831                    returnValue = new Timestamp(lval);
832
833                } else if (targetType.equals(Integer.class) || targetType.equals(int.class)) {
834                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is INT and column is LONG");
835                    returnValue = Integer.parseInt("" + lval);
836
837                } else if (targetType.equals(LocalDate.class)) {
838                    // SQLite reads long as date.....
839                    returnValue = new Timestamp(lval).toLocalDateTime().toLocalDate();
840
841                } else if (targetType.equals(LocalDateTime.class)) {
842                    returnValue = new Timestamp(lval).toLocalDateTime();
843
844                } else if (targetType.equals(Time.class)) {
845                    // SQLite.... Again.....
846                    returnValue = new Time((Long) value);
847                }
848                break;
849
850            case floatType:
851            case FloatType:
852                log.debug("FloatType");
853                break;
854
855            case doubleType:
856            case DoubleType:
857                Double dbl = (Double) value;
858                // float or doubles to BigDecimal
859                if (targetType.equals(BigDecimal.class)) {
860                    returnValue = new BigDecimal("" + value);
861
862                } else if (targetType.equals(Float.class) || targetType.equals(float.class)) {
863                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is FLOAT and column is DOUBLE");
864                    returnValue = dbl.floatValue();
865
866                } else if (targetType.equals(Integer.class) || targetType.equals(int.class)) {
867                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is INT and column is DOUBLE");
868                    returnValue = dbl.intValue();
869
870                } else if (targetType.equals(Long.class) || targetType.equals(long.class)) {
871                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Long and column is DOUBLE");
872                    returnValue = dbl.longValue();
873                }
874                break;
875
876            case BigDecimalType:
877                if (targetType.equals(Float.class) || targetType.equals(float.class)) {
878                    returnValue = ((Number) value).floatValue();
879                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Float and column is BigDecimal");
880
881                } else if (targetType.equals(Double.class) || targetType.equals(double.class)) {
882                    returnValue = ((Number) value).doubleValue();
883                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Double and column is BigDecimal");
884
885                } else if (targetType.equals(Long.class) || targetType.equals(long.class)) {
886                    returnValue = ((Number) value).longValue();
887                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Long and column is BigDecimal");
888
889                } else if (targetType.equals(Integer.class) || targetType.equals(int.class)) {
890                    returnValue = ((Number) value).intValue();
891                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Integer and column is BigDecimal");
892
893                } else if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
894                    // BigDecimal to Boolean. Oracle (sigh) - Additional for a Char to Boolean as then (see TestOracle for links)
895                    returnValue = ((Number) value).intValue() == 1;
896                    warnNoDuplicates("Possible overflow column " + columnName + " - Property is Boolean and column is BigDecimal - seems a bit overkill?");
897
898                } else if (targetType.equals(String.class)) {
899                    returnValue = (value).toString();
900                }
901                break;
902
903            case StringType:
904                java.util.Date dval;
905                String format;
906                if (("" + value).length() > "yyyy-MM-dd".length()) {
907                    format = "yyyy-MM-dd hh:mm:ss";
908                } else {
909                    format = "yyyy-MM-dd";
910                }
911                DateFormat df = new SimpleDateFormat(format);
912
913                // Read a string but we want a date
914                if (targetType.equals(java.util.Date.class) || targetType.equals(java.sql.Date.class)) {
915                    // This condition occurs in SQLite when you have a datetime with default annotated
916                    // the format returned is 2012-06-02 19:59:49
917                    // Used for SQLite returning dates as Strings under some conditions
918                    // SQL or others may return STRING yyyy-MM-dd for older legacy 'date' type.
919                    // https://docs.microsoft.com/en-us/sql/t-sql/data-types/date-transact-sql?view=sql-server-ver15
920                    dval = tryParseDate(value, targetType, columnName, df);
921
922                    if (targetType.equals(java.sql.Date.class)) {
923                        returnValue = new java.sql.Date(dval.getTime());
924                    } else {
925                        returnValue = dval;
926                    }
927
928                } else if (targetType.equals(Timestamp.class)) {
929                    returnValue = tryParseTimestamp(value, targetType, columnName);
930
931                } else if (targetType.equals(LocalDate.class)) {
932                    // JTDS
933                    dval = tryParseDate(value, targetType, columnName, df);
934                    returnValue = dval.toInstant().atZone(ZoneId.systemDefault()).toLocalDate();
935
936                } else if (targetType.equals(LocalDateTime.class)) {
937                    // JTDS
938                    dval = tryParseDate(value, targetType, columnName, df);
939                    returnValue = dval.toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime();
940
941                } else if (targetType.isEnum()) {
942                    // If this is an enum do a case insensitive comparison
943                    Object[] enumConstants = targetType.getEnumConstants();
944                    for (Object element : enumConstants) {
945                        if (("" + value).equalsIgnoreCase(element.toString())) {
946                            returnValue = element;
947                            break;
948                        }
949                    }
950
951                } else if (targetType.equals(UUID.class)) {
952                    returnValue = UUID.fromString("" + value);
953
954                } else if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
955                    // String to Boolean - true or 1 - otherwise false (or null)
956                    String bval = "" + value;
957                    returnValue = bval.equalsIgnoreCase("true") || bval.equals("1");
958
959                } else if (targetType.equals(Time.class)) {
960                    // MSSQL works, JTDS returns Varchar in format below with varying decimal numbers
961                    // which won't format unless I use Exact so I chop of the milliseconds.
962                    DateFormat timeFormat = new SimpleDateFormat("hh:mm:ss");
963                    String sval = "" + value;
964                    if (sval.indexOf('.') > -1) {
965                        sval = sval.substring(0, sval.indexOf('.'));
966                    }
967                    dval = tryParseDate(sval, targetType, columnName, timeFormat);
968                    returnValue = new Time(dval.getTime());
969
970                } else if (targetType.equals(LocalTime.class)) {
971                    // JTDS Fails again...
972                    returnValue = LocalTime.parse("" + value);
973
974                } else if (targetType.equals(Character.class) || targetType.equals(char.class)) {
975                    String s = "" + value;
976                    if (s.length() > 0) {
977                        returnValue = s.charAt(0);
978                    }
979
980                } else if (targetType.equals(BigDecimal.class)) {
981                    try {
982                        returnValue = new BigDecimal("" + value);
983                    } catch (NumberFormatException e) {
984                        String msg = "NumberFormatException: Column: " + columnName + " Type of property: " + targetType + " - Type read: " + value.getClass() + " VALUE: " + value;
985                        throw new PersismException(msg, e);
986                    }
987                }
988                break;
989
990            case characterType:
991            case CharacterType:
992                log.debug("CharacterType");
993                break;
994
995            case LocalDateType:
996                log.debug("LocalDateType");
997                returnValue = java.sql.Date.valueOf((LocalDate) value);
998                break;
999
1000            case LocalDateTimeType:
1001                log.debug("LocalDateTimeType");
1002                returnValue = Timestamp.valueOf((LocalDateTime) value);
1003                break;
1004
1005            case LocalTimeType:
1006                log.debug("LocalTimeType");
1007                returnValue = Time.valueOf((LocalTime) value);
1008                break;
1009
1010            case UtilDateType:
1011            case SQLDateType:
1012            case TimestampType:
1013                if (targetType.equals(java.util.Date.class)) {
1014                    returnValue = new java.util.Date(((Date) value).getTime());
1015
1016                } else if (targetType.equals(java.sql.Date.class)) {
1017                    returnValue = new java.sql.Date(((Date) value).getTime());
1018
1019                } else if (targetType.equals(Timestamp.class)) {
1020                    returnValue = new Timestamp(((Date) value).getTime());
1021
1022                } else if (targetType.equals(LocalDate.class)) {
1023                    Date dt = new Date(((Date) value).getTime());
1024                    returnValue = dt.toInstant().atZone(ZoneId.systemDefault()).toLocalDate();
1025
1026                } else if (targetType.equals(LocalDateTime.class)) {
1027                    Date dt = new Date(((Date) value).getTime());
1028                    returnValue = dt.toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime();
1029
1030                } else if (targetType.equals(Time.class)) {
1031                    // Oracle doesn't seem to have Time so we use Timestamp
1032                    returnValue = new Time(((Date) value).getTime());
1033
1034                } else if (targetType.equals(LocalTime.class)) {
1035                    // Oracle.... Sigh
1036                    Date dt = (Date) value;
1037                    returnValue = dt.toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime().toLocalTime();
1038                }
1039                break;
1040
1041            case TimeType:
1042                log.debug("TimeType");
1043                if (targetType.equals(LocalTime.class)) {
1044                    returnValue = LocalTime.parse("" + value);
1045                }
1046                break;
1047
1048            case InstantType:
1049            case OffsetDateTimeType:
1050            case ZonedDateTimeType:
1051                log.warn(valueType + " not yet supported", new Throwable());
1052                break;
1053
1054            case byteArrayType:
1055            case ByteArrayType:
1056                log.debug("ByteArrayType");
1057                if (targetType.equals(UUID.class)) {
1058                    returnValue = Util.asUuid((byte[]) value);
1059                }
1060                break;
1061
1062            case ClobType:
1063            case BlobType:
1064                log.warn("why? Clob is read as String, Blob as byte array - see readColumn method", new Throwable());
1065                break;
1066
1067            case EnumType:
1068                // No need to convert it here.
1069                // If it's being used for the property setter then it's OK
1070                // If it's being used by setParameters it's converted to String
1071                // The String case above converts from the String to the Enum
1072                log.debug("EnumType");
1073                break;
1074
1075            case UUIDType:
1076                log.debug("UUIDType");
1077                if (targetType.equals(Blob.class) || targetType.equals(byte[].class) || targetType.equals(Byte[].class)) {
1078                    returnValue = Util.asBytes((UUID) value);
1079                }
1080                break;
1081
1082            case ObjectType:
1083                log.debug("ObjectType");
1084                break;
1085        }
1086        return returnValue;
1087    }

1111
1112        Object value = null;
1113        Types type = Types.getType(objectClass);
1114
1115        if (type == null) {
1116            log.warn("Unhandled type " + objectClass);
1117            return (T) rs.getObject(1);
1118        }
1119
1120        switch (type) {
1121
1122            case integerType:
1123            case IntegerType:
1124                value = rs.getInt(1);
1125                break;
1126
1127            case longType:
1128            case LongType:
1129                value = rs.getLong(1);
1130                break;
1131
1132            default:
1133                value = rs.getObject(1);
1134        }
1135        return (T) value;

1138    void setParameters(PreparedStatement st, Object[] parameters) throws SQLException {
1139        if (log.isDebugEnabled()) {
1140            log.debug("PARAMS: " + Arrays.asList(parameters));
1141        }
1142
1143        int n = 1;
1144        for (Object param : parameters) {
1145
1146            if (param != null) {
1147
1148                Types type = Types.getType(param.getClass());
1149                assert type != null;
1150
1151                switch (type) {
1152
1153                    case booleanType:
1154                    case BooleanType:
1155                        st.setBoolean(n, (Boolean) param);
1156                        break;
1157
1158                    case byteType:
1159                    case ByteType:
1160                        st.setByte(n, (Byte) param);
1161                        break;
1162
1163                    case shortType:
1164                    case ShortType:
1165                        st.setShort(n, (Short) param);
1166                        break;
1167
1168                    case integerType:
1169                    case IntegerType:
1170                        st.setInt(n, (Integer) param);
1171                        break;
1172
1173                    case longType:
1174                    case LongType:
1175                        st.setLong(n, (Long) param);
1176                        break;
1177
1178                    case floatType:
1179                    case FloatType:
1180                        st.setFloat(n, (Float) param);
1181                        break;
1182
1183                    case doubleType:
1184                    case DoubleType:
1185                        st.setDouble(n, (Double) param);
1186                        break;
1187
1188                    case BigDecimalType:
1189                        st.setBigDecimal(n, (BigDecimal) param);
1190                        break;
1191
1192                    case BigIntegerType:
1193                        st.setString(n, "" + param);
1194                        break;
1195
1196                    case StringType:
1197                        st.setString(n, (String) param);
1198                        break;
1199
1200                    case characterType:
1201                    case CharacterType:
1202                        st.setObject(n, "" + param);
1203                        break;
1204
1205                    case SQLDateType:
1206                        st.setDate(n, (java.sql.Date) param);
1207                        break;
1208
1209                    case TimeType:
1210                        st.setTime(n, (Time) param);
1211                        break;
1212
1213                    case TimestampType:
1214                        st.setTimestamp(n, (Timestamp) param);
1215                        break;
1216
1217                    // THESE are converted to Timestamp (or Time) by convert method.
1218                    case LocalTimeType:
1219                    case LocalDateType:
1220                    case LocalDateTimeType:
1221                        log.warn(type + " why would this occur in setParameters?", new Throwable());
1222                        break;
1223
1224                    case OffsetDateTimeType:
1225                    case ZonedDateTimeType:
1226                    case InstantType:
1227                        // todo ZonedDateTime, OffsetDateTimeType and MAYBE Instant
1228                        break;
1229
1230                    case byteArrayType:
1231                    case ByteArrayType:
1232                        // Blob maps to byte array
1233                        st.setBytes(n, (byte[]) param);
1234                        break;
1235
1236                    case ClobType:
1237                    case BlobType:
1238                        // Clob is converted to String Blob is converted to byte array
1239                        // So this should not occur.
1240                        log.warn(type + " why would this occur in setParameters?", new Throwable());
1241                        break;
1242
1243                    case EnumType:
1244                        if (metaData.connectionType == ConnectionTypes.PostgreSQL) {
1245                            st.setObject(n, param.toString(), java.sql.Types.OTHER);
1246                        } else {
1247                            st.setString(n, param.toString());
1248                        }
1249                        break;
1250
1251                    case UUIDType:
1252                        if (metaData.connectionType == ConnectionTypes.PostgreSQL) {
1253                            // PostgreSQL does work with setObject but not setString unless you set the connection property stringtype=unspecified
1254                            st.setObject(n, param);
1255                        } else {
1256                            st.setString(n, param.toString());
1257                        }
1258                        break;
1259
1260                    default:
1261                        // Usually SQLite with util.date - setObject works
1262                        log.info("setParameters default: " + n + " " + param.getClass());
1263                        st.setObject(n, param);
1264                }
1265
1266            } else {
1267                // param is null
1268                st.setObject(n, param);
1269            }
1270
1271            n++;
1272        }
1273    }
