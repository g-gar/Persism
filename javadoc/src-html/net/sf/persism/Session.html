055    /**
056     * Updates the data object in the database.
057     *
058     * @param object data object to update.
059     * @return usually 1 to indicate rows changed via JDBC.
060     * @throws PersismException Indicating the upcoming robot uprising.
061     */
062    public int update(Object object) throws PersismException {
063        List<String> primaryKeys = metaData.getPrimaryKeys(object.getClass(), connection);
064        if (primaryKeys.size() == 0) {
065            throw new PersismException("Cannot perform update. " + metaData.getTableName(object.getClass()) + " has no primary keys.");
066        }
067
068        PreparedStatement st = null;
069        try {
070
071            String updateStatement = null;
072            try {
073                updateStatement = metaData.getUpdateStatement(object, connection);
074            } catch (NoChangesDetectedForUpdateException e) {
075                log.info("No properties changed. No update required for Object: " + object + " class: " + object.getClass().getName());
076                return 0;
077            }
078
079            st = connection.prepareStatement(updateStatement);
080
081            // These keys should always be in sorted order.
082            Map<String, PropertyInfo> allProperties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
083            Map<String, PropertyInfo> changedProperties;
084            if (object instanceof Persistable) {
085                changedProperties = metaData.getChangedProperties((Persistable) object, connection);
086            } else {
087                changedProperties = allProperties;
088            }
089
090            List<Object> params = new ArrayList<>(primaryKeys.size());
091            for (String column : changedProperties.keySet()) {
092                if (!primaryKeys.contains(column)) {
093                    Object value = allProperties.get(column).getter.invoke(object);
094
095                    if (value != null) {
096                        if (value instanceof java.util.Date) {
097                            java.util.Date dt = (java.util.Date) value;
098                            value = new Timestamp(dt.getTime());
099                        }
100
101                        if (value instanceof String) {
102                            // check width
103                            ColumnInfo columnInfo = metaData.getColumns(object.getClass(), connection).get(column);
104                            String str = (String) value;
105                            if (str.length() > columnInfo.length) {
106                                str = str.substring(0, columnInfo.length);
107                                // todo Should Persism strict should throw this as an exception?
108                                log.warn("TRUNCATION with Column: " + column + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
109                                value = str;
110                            }
111                        }
112                    }
113                    params.add(value);
114                }
115            }
116
117            for (String column : primaryKeys) {
118                params.add(allProperties.get(column).getter.invoke(object));
119            }
120            setParameters(st, params.toArray());
121            int ret = st.executeUpdate();
122            return ret;
123
124        } catch (Exception e) {
125            Util.rollback(connection);
126            throw new PersismException(e.getMessage(), e);
127
128        } finally {
129            Util.cleanup(st, null);
130        }
131    }

133    /**
134     * Inserts the data object in the database refreshing with autoinc and other defaults that may exist.
135     *
136     * @param object the data object to insert.
137     * @return usually 1 to indicate rows changed via JDBC.
138     * @throws PersismException When planet of the apes starts happening.
139     */
140    public int insert(Object object) throws PersismException {
141        String insertStatement = metaData.getInsertStatement(object, connection);
142
143        PreparedStatement st = null;
144        ResultSet rs = null;
145
146        try {
147            // These keys should always be in sorted order.
148            Map<String, PropertyInfo> properties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
149            Map<String, ColumnInfo> columns = metaData.getColumns(object.getClass(), connection);
150
151            List<String> generatedKeys = new ArrayList<>(1);
152            for (ColumnInfo column : columns.values()) {
153                if (column.autoIncrement) {
154                    generatedKeys.add(column.columnName);
155                }
156            }
157
158            if (generatedKeys.size() > 0) {
159                String[] keyArray = generatedKeys.toArray(new String[0]);
160                st = connection.prepareStatement(insertStatement, keyArray);
161            } else {
162                st = connection.prepareStatement(insertStatement);
163            }
164
165            boolean tableHasDefaultColumnValues = false;
166
167            List<Object> params = new ArrayList<>();
168            for (ColumnInfo columnInfo : columns.values()) {
169
170                PropertyInfo propertyInfo = properties.get(columnInfo.columnName);
171                if (!columnInfo.autoIncrement) {
172
173                    // See MetaData getInsertStatement - Maybe we should return a new Object type for InsertStatement
174                    if (columnInfo.hasDefault) {
175                        // Do not include if this column has a default and no value has been
176                        // set on it's associated property.
177                        if (properties.get(columnInfo.columnName).getter.invoke(object) == null) {
178
179                            if (columnInfo.primary) {
180                                throw new PersismException("Non-auto inc generated primary keys are not supported. Please assign your primary key value before performing an insert.");
181                            }
182
183                            tableHasDefaultColumnValues = true;
184                            continue;
185                        }
186                    }
187
188                    Object value = propertyInfo.getter.invoke(object);
189
190                    if (value != null) {
191                        // sql.Date is a subclass so this would be true
192                        // same with sql.Time
193                        if (value instanceof java.util.Date) {
194                            java.util.Date dt = (java.util.Date) value;
195                            value = new Timestamp(dt.getTime());
196                        }
197
198                        if (value instanceof String) {
199                            // check width
200                            String str = (String) value;
201                            if (str.length() > columnInfo.length) {
202                                // todo should Persism strict throw this as an exception?
203                                str = str.substring(0, columnInfo.length);
204                                log.warn("TRUNCATION with Column: " + columnInfo.columnName + " for table: " + metaData.getTableName(object.getClass()) + ". Old value: \"" + value + "\" New value: \"" + str + "\"");
205                                value = str;
206                            }
207                        }
208                    }
209                    params.add(value);
210                }
211            }
212
213            // https://forums.oracle.com/forums/thread.jspa?threadID=879222
214            // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
215            //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
216            setParameters(st, params.toArray());
217            int ret = st.executeUpdate();
218            if (log.isDebugEnabled()) {
219                log.debug("insert ret: " + ret);
220            }
221
222            if (generatedKeys.size() > 0) {
223                rs = st.getGeneratedKeys();
224                for (String column : generatedKeys) {
225                    if (rs.next()) {
226
227                        Method setter = properties.get(column).setter;
228
229                        if (setter != null) {
230                            Object value = getTypedValueReturnedFromGeneratedKeys(setter.getParameterTypes()[0], rs);
231
232                            if (log.isDebugEnabled()) {
233                                log.debug(column + " generated " + value); // HERE!
234                                log.debug(setter);
235                            }
236                            setter.invoke(object, value);
237
238                        } else {
239                            log.warn("no setter found for column " + column);
240                        }
241                    }
242                }
243            }
244
245            if (tableHasDefaultColumnValues) {
246                // Read the full object back to update any properties which had defaults
247                fetch(object);
248            }
249
250            return ret;
251        } catch (Exception e) {
252            Util.rollback(connection);
253            throw new PersismException(e.getMessage(), e);
254        } finally {
255            Util.cleanup(st, rs);
256        }
257    }

326    /**
327     * Query for a list of objects of the specified class using the specified SQL query and parameters.
328     * The type of the list can be Data Objects or native Java Objects or primitives.
329     *
330     * @param objectClass class of objects to return.
331     * @param sql         query string to execute.
332     * @param parameters  parameters to the query.
333     * @param <T>         Return type
334     * @return a list of objects of the specified class using the specified SQL query and parameters.
335     * @throws PersismException If something goes wrong you get a big stack trace.
336     */
337    public <T> List<T> query(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
338        List<T> list = new ArrayList<T>(32);
339
340        Result result = new Result();
341
342        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
343        boolean readPrimitive = Types.getType(objectClass) != null;
344
345        if (!readPrimitive && objectClass.getAnnotation(NotTable.class) == null) {
346            // Make sure columns are initialized if this is a table.
347            metaData.getTableColumnsPropertyInfo(objectClass, connection);
348        }
349
350        try {
351
352            exec(result, sql, parameters);
353
354            while (result.rs.next()) {
355
356                if (readPrimitive) {
357                    list.add((T) readPrimitive(result.rs, 1, objectClass));
358                } else {
359                    // should be getDeclaredConstructor().newInstance() now.
360                    T t = objectClass.newInstance();
361                    t = (T) readObject(t, result.rs);
362                    list.add(t);
363                }
364            }
365
366        } catch (Exception e) {
367            Util.rollback(connection);
368            throw new PersismException(e.getMessage(), e);
369        } finally {
370            Util.cleanup(result.st, result.rs);
371        }
372
373        return list;
374
375    }

377    /**
378     * Fetch an object from the database by it's primary key(s).
379     * You should instantiate the object and set the primary key properties before calling this method.
380     *
381     * @param object Data object to read from the database.
382     * @return true if the object was found by the primary key.
383     * @throws PersismException if something goes wrong.
384     */
385    public boolean fetch(Object object) throws PersismException {
386        Class objectClass = object.getClass();
387
388        // If we know this type it means it's a primitive type. This method cannot be used for primitives
389        boolean readPrimitive = Types.getType(objectClass) != null;
390        if (readPrimitive) {
391            throw new PersismException("Cannot read a primitive type object with this method.");
392        }
393
394        List<String> primaryKeys = metaData.getPrimaryKeys(objectClass, connection);
395        if (primaryKeys.size() == 0) {
396            throw new PersismException("Cannot perform readObjectByPrimary. " + metaData.getTableName(objectClass) + " has no primary keys.");
397        }
398
399        Map<String, PropertyInfo> properties = metaData.getTableColumnsPropertyInfo(object.getClass(), connection);
400        List<Object> params = new ArrayList<>(primaryKeys.size());
401
402        Result result = new Result();
403        try {
404            for (String column : primaryKeys) {
405                PropertyInfo propertyInfo = properties.get(column);
406                params.add(propertyInfo.getter.invoke(object));
407            }
408            String sql = metaData.getSelectStatement(object, connection);
409            if (log.isDebugEnabled()) {
410                log.debug("FETCH " + sql + " PARAMS: " + params);
411            }
412            exec(result, sql, params.toArray());
413
414            if (result.rs.next()) {
415                readObject(object, result.rs);
416                return true;
417            }
418            return false;
419
420        } catch (Exception e) {
421            Util.rollback(connection);
422            throw new PersismException(e.getMessage(), e);
423        } finally {
424            Util.cleanup(result.st, result.rs);
425        }
426    }

449    /**
450     * Fetch an object of the specified type from the database. The type can be a Data Object or a native Java Object or primitive.
451     *
452     * @param objectClass Type of returned value
453     * @param sql         query - this would usually be a select OR a select of a single column if the type is a primitive.
454     *                    If this is a primitive type then this method will only look at the 1st column in the result.
455     * @param parameters  parameters to the query.
456     * @param <T>         Return type
457     * @return value read from the database of type T or null if not found
458     * @throws PersismException Well, this is a runtime exception so actually it could be anything really.
459     */
460    public <T> T fetch(Class<T> objectClass, String sql, Object... parameters) throws PersismException {
461        // If we know this type it means it's a primitive type. Not a DAO so we use a different rule to read those
462        boolean readPrimitive = Types.getType(objectClass) != null;
463
464        Result result = new Result();
465        try {
466
467            exec(result, sql, parameters);
468
469            if (result.rs.next()) {
470
471                if (readPrimitive) {
472                    return (T) readPrimitive(result.rs, 1, objectClass);
473
474                } else {
475                    T t = objectClass.newInstance();
476                    readObject(t, result.rs);
477                    return t;
478                }
479            }
480
481            return null;
482
483        } catch (Exception e) {
484            Util.rollback(connection);
485            throw new PersismException(e.getMessage(), e);
486        } finally {
487            Util.cleanup(result.st, result.rs);
488        }
489    }

499    private Result exec(Result result, String sql, Object... parameters) throws SQLException {
500        if (sql.toLowerCase().startsWith("select ")) {
501            result.st = connection.prepareStatement(sql);
502
503            PreparedStatement pst = (PreparedStatement) result.st;
504            setParameters(pst, parameters);
505            result.rs = pst.executeQuery();
506        } else {
507            if (!sql.toLowerCase().startsWith("{call")) {
508                sql = "{call " + sql + "} ";
509            }
510            result.st = connection.prepareCall(sql);
511
512            CallableStatement cst = (CallableStatement) result.st;
513            setParameters(cst, parameters);
514            result.rs = cst.executeQuery();
515        }
516        return result;
517    }

519    private <T> T readObject(Object object, ResultSet rs) throws IllegalAccessException, SQLException, InvocationTargetException, IOException {
520
521        Class objectClass = object.getClass();
522        // We should never call this method with a primitive type.
523        assert Types.getType(objectClass) == null;
524
525        Map<String, PropertyInfo> properties;
526        if (objectClass.getAnnotation(NotTable.class) == null) {
527            properties = metaData.getTableColumnsPropertyInfo(objectClass, connection);
528        } else {
529            properties = metaData.getQueryColumnsPropertyInfo(objectClass, rs);
530        }
531
532        // Test if all properties have column mapping and throw PersismException if not
533        // This block verifies that the object is fully initialized.
534        // Any properties not marked by NotMapped should have been set (or if they have a getter only)
535        // If not throw a PersismException
536        Collection<PropertyInfo> allProperties = MetaData.getPropertyInfo(objectClass);
537        if (properties.values().size() < allProperties.size()) {
538            Set<PropertyInfo> missing = new HashSet<PropertyInfo>(allProperties.size());
539            missing.addAll(allProperties);
540            missing.removeAll(properties.values());
541
542            StringBuilder sb = new StringBuilder();
543            String sep = "";
544            for (PropertyInfo prop : missing) {
545                sb.append(sep).append(prop.propertyName);
546                sep = ",";
547            }
548
549            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not initialized in the queried columns (" + sb + ").");
550        }
551        ResultSetMetaData rsmd = rs.getMetaData();
552        int columnCount = rsmd.getColumnCount();
553        List<String> foundColumns = new ArrayList<String>(columnCount);
554
555        for (int j = 1; j <= columnCount; j++) {
556
557            String columnName = rsmd.getColumnLabel(j);
558            PropertyInfo columnProperty = properties.get(columnName);
559
560            if (columnProperty != null) {
561                Class getterType = columnProperty.getter.getReturnType();
562
563                Object value = readPrimitive(rs, j, getterType);
564
565                foundColumns.add(columnName);
566
567                if (value != null) {
568                    try {
569                        columnProperty.setter.invoke(object, value);
570                    } catch (IllegalArgumentException e) {
571                        String msg = e.getMessage() + " Object " + objectClass + ". Column: " + columnName + " Type of property: " + getterType + " - Type read: " + value.getClass() + " VALUE: " + value;
572                        throw new PersismException(msg, e);
573                    }
574
575                }
576            }
577        }
578
579        // This is doing a similar check to above but on the ResultSet itself.
580        // This tests for when a user writes their own SQL and forgets a column.
581        if (foundColumns.size() < properties.keySet().size()) {
582
583            Set<String> missing = new HashSet<String>(columnCount);
584            missing.addAll(properties.keySet());
585            missing.removeAll(foundColumns);
586
587            // todo maybe strict mode off logs warn? Should we do this if this is Query vs Table?
588            throw new PersismException("Object " + objectClass + " was not properly initialized. Some properties not initialized by the queried columns: " + foundColumns + " Missing:" + missing);
589        }
590
591        if (object instanceof Persistable) {
592            // Save this object's initial state to later detect changed properties
593            ((Persistable) object).saveReadState();
594        }
595
596        return (T) object;
597
598    }

600    private Object readPrimitive(ResultSet rs, int column, Class returnType) throws SQLException, IOException {
601        ResultSetMetaData rsmd = rs.getMetaData();
602        int sqlColumnType = rsmd.getColumnType(column);
603        String columnName = rsmd.getColumnLabel(column);
604        Object value = read(rs, column, columnName, sqlColumnType, returnType);
605        return value;
606    }

608    // Make an educated guess read an object from the ResultSet to get the best Class type of the object
609    private Object read(ResultSet rs, int column, String columnName, int sqlColumnType, Class returnType) throws SQLException, IOException {
610        Object value;
611        Types columnType = Types.convert(sqlColumnType); // note this could be null if we can't match a type
612        if (columnType != null) {
613
614            switch (columnType) {
615
616                case TimestampType:
617                    // work around to Oracle reading a oracle.sql.TIMESTAMP class with getObject
618                    value = rs.getTimestamp(column);
619                    break;
620                case ClobType:
621                    Clob clob = rs.getClob(column);
622                    try (InputStream in = clob.getAsciiStream()) {
623                        StringWriter writer = new StringWriter();
624
625                        int c = -1;
626                        while ((c = in.read()) != -1) {
627                            writer.write(c);
628                        }
629                        writer.flush();
630                        value = writer.toString();
631                    }
632                    break;
633                case BlobType:
634                    byte[] buffer = new byte[1024];
635                    Blob blob = rs.getBlob(column);
636                    try (InputStream in = blob.getBinaryStream()) {
637                        // if it goes past int length the oh well...
638                        ByteArrayOutputStream bos = new ByteArrayOutputStream((int) blob.length());
639                        for (int len; (len = in.read(buffer)) != -1; ) {
640                            bos.write(buffer, 0, len);
641                        }
642                        value = bos.toByteArray();
643                    }
644                    break;
645
646                case IntegerType:
647                    value = rs.getObject(column) == null ? null : rs.getInt(column);
648                    break;
649                case LongType:
650                    value = rs.getObject(column) == null ? null : rs.getLong(column);
651                    break;
652                case FloatType:
653                    value = rs.getObject(column) == null ? null : rs.getFloat(column);
654                    break;
655                case DoubleType:
656                    value = rs.getObject(column) == null ? null : rs.getDouble(column);
657                    break;
658                case DecimalType:
659                    value = rs.getBigDecimal(column);
660                    break;
661                case TimeType:
662                    value = rs.getTime(column);
663                    break;
664// We can't assume rs.getDate will work. SQLITE actually has a long value in here.
665// We can live with rs.getObject and the convert method will handle it.
666//                case SQLDateType:
667//                case UtilDateType:DateType:
668//                    value = rs.getDate(column);
669//                    break;
670                default:
671                    value = rs.getObject(column);
672            }
673
674        } else {
675            log.warn("Column type not known for SQL type " + sqlColumnType, new Throwable());
676            value = rs.getObject(column);
677        }
678
679        // If value is null or column type is unknown - no need to try to convert anything.
680        if (value != null && columnType != null) {
681            value = convert(value, returnType, columnName);
682        }
683
684        return value;
685    }

687    // Make a sensible conversion of the Value read from the DB and the property type defined on the Data class.
688    private Object convert(Object dbValue, Class propertyType, String columnName) {
689        Types valueType = Types.getType(dbValue.getClass());
690        if (valueType == null) {
691            log.error("WTF " + dbValue.getClass());
692        }
693
694        // try to convert or cast the value to the proper type.
695        switch (valueType) {
696
697            case booleanType:
698            case BooleanType:
699                log.debug("BooleanType");
700                break;
701
702            case byteType:
703            case ByteType:
704            case shortType:
705            case ShortType:
706            case integerType:
707            case IntegerType:
708                // int to bool
709                if (propertyType == Boolean.class || propertyType == boolean.class) {
710                    dbValue = (Integer.valueOf("" + dbValue) == 0) ? false : true;
711                }
712                break;
713
714            case longType:
715            case LongType:
716                // long to date
717                if (propertyType.equals(java.util.Date.class) || propertyType.equals(java.sql.Date.class)) {
718                    long lval = Long.valueOf("" + dbValue);
719
720                    if (propertyType.equals(java.sql.Date.class)) {
721                        dbValue = new java.sql.Date(lval);
722                    } else {
723                        dbValue = new java.util.Date(lval);
724                    }
725
726                } else if (propertyType.equals(Integer.class) || propertyType.equals(int.class)) {
727                    warnOverflow("Possible overflow column " + columnName + " - Property is INT and column value is LONG");
728                    dbValue = Integer.parseInt("" + dbValue);
729
730                } else if (propertyType.equals(LocalDate.class) ) {
731                    // SQLite reads long as date.....
732                    dbValue = new Timestamp((long)dbValue).toLocalDateTime().toLocalDate();
733
734                } else if (propertyType.equals(LocalDateTime.class)) {
735                    dbValue = new Timestamp((long)dbValue).toLocalDateTime();
736                }
737
738                break;
739
740            case floatType:
741            case FloatType:
742                log.debug("FloatType");
743                break;
744
745            case doubleType:
746            case DoubleType:
747                // float or doubles to BigDecimal
748                if (propertyType == BigDecimal.class) {
749                    dbValue = new BigDecimal("" + dbValue);
750                } else if (propertyType == Float.class || propertyType == float.class) {
751                    warnOverflow("Possible overflow column " + columnName + " - Property is FLOAT and column value is DOUBLE");
752                    dbValue = Float.parseFloat("" + dbValue);
753                } else if (propertyType == Integer.class || propertyType == int.class) {
754                    warnOverflow("Possible overflow column " + columnName + " - Property is INT and column value is DOUBLE");
755                    String val = "" + dbValue;
756                    if (val.contains(".")) {
757                        val = val.substring(0, val.indexOf("."));
758                    }
759                    dbValue = Integer.parseInt(val);
760                }
761                break;
762
763            case DecimalType:
764                if (propertyType == Float.class || propertyType == float.class) {
765                    dbValue = ((BigDecimal) dbValue).floatValue();
766                    warnOverflow("Possible overflow column " + columnName + " - Property is Float and column value is BigDecimal");
767                } else if (propertyType == Double.class || propertyType == double.class) {
768                    dbValue = ((BigDecimal) dbValue).doubleValue();
769                    warnOverflow("Possible overflow column " + columnName + " - Property is Double and column value is BigDecimal");
770                } else if (propertyType == Long.class || propertyType == long.class) {
771                    dbValue = ((BigDecimal) dbValue).longValue();
772                    warnOverflow("Possible overflow column " + columnName + " - Property is Long and column value is BigDecimal");
773                } else if (propertyType == Integer.class || propertyType == int.class) {
774                    dbValue = ((BigDecimal) dbValue).intValue();
775                    warnOverflow("Possible overflow column " + columnName + " - Property is Integer and column value is BigDecimal");
776                } else if (propertyType == Boolean.class || propertyType == boolean.class) {
777                    // BigDecimal to Boolean. Oracle (sigh) - Additional for a Char to Boolean as then (see TestOracle for links)
778                    dbValue = ((BigDecimal) dbValue).intValue() == 1;
779                    warnOverflow("Possible overflow column " + columnName + " - Property is Boolean and column value is BigDecimal - seems a bit overkill?");
780                } else if (propertyType == String.class) {
781                    dbValue = (dbValue).toString();
782                }
783                break;
784
785            case StringType:
786
787                // Read a string but we want a date
788                if (propertyType.equals(java.util.Date.class) || propertyType.equals(java.sql.Date.class)) {
789                    // This condition occurs in SQLite when you have a datetime with default annotated
790                    // the format returned is 2012-06-02 19:59:49
791                    java.util.Date dval = null;
792                    try {
793                        // Used for SQLite returning dates as Strings under some conditions
794                        // SQL or others may return STRING yyyy-MM-dd for older legacy 'date' type.
795                        // https://docs.microsoft.com/en-us/sql/t-sql/data-types/date-transact-sql?view=sql-server-ver15
796                        String format;
797                        if (("" + dbValue).length() > "yyyy-MM-dd".length()) {
798                            format = "yyyy-MM-dd hh:mm:ss";
799                        } else {
800                            format = "yyyy-MM-dd";
801                        }
802                        DateFormat df = new SimpleDateFormat(format);
803                        dval = df.parse("" + dbValue);
804                    } catch (ParseException e) {
805                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + propertyType + " - Type read: " + dbValue.getClass() + " VALUE: " + dbValue;
806                        throw new PersismException(msg, e);
807                    }
808
809                    if (propertyType.equals(java.sql.Date.class)) {
810                        dbValue = new java.sql.Date(dval.getTime());
811                    } else {
812                        dbValue = dval;
813                    }
814
815                } else if (propertyType.equals(java.sql.Timestamp.class)) {
816                    // String to timestamp?
817                    // value = new Timestamp()
818                    java.util.Date dval = null;
819                    DateFormat df = new SimpleDateFormat("yyyy-MM-dd hh:mm:ss");
820                    try {
821                        dval = df.parse("" + dbValue);
822                        dbValue = new Timestamp(dval.getTime());
823                    } catch (ParseException e) {
824                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + propertyType + " - Type read: " + dbValue.getClass() + " VALUE: " + dbValue;
825                        throw new PersismException(msg, e);
826                    }
827                } else if (propertyType.equals(LocalDate.class)) {
828                    DateFormat df = new SimpleDateFormat("yyyy-MM-dd");
829                    java.util.Date dval = null;
830                    try {
831                        dval = df.parse("" + dbValue);
832                        dbValue = Instant.ofEpochMilli(dval.getTime())
833                                .atZone(ZoneId.systemDefault())
834                                .toLocalDate();
835                    } catch (ParseException e) {
836                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + propertyType + " - Type read: " + dbValue.getClass() + " VALUE: " + dbValue;
837                        throw new PersismException(msg, e);
838                    }
839// This block should maybe have occurred because SQLite usually reads dates as strings
840// but for whatever reason getObject call in the read() method returns LONG so we
841// never get here. The convert to LocalDateTime is made under the Long case
842//                } else if (propertyType.equals(LocalDateTime.class)) {
843//                    DateFormat df = new SimpleDateFormat("yyyy-MM-dd hh:mm:ss");
844//                    java.util.Date dval = null;
845//                    try {
846//                        dval = df.parse("" + dbValue);
847//                        dbValue = Instant.ofEpochMilli(dval.getTime())
848//                                .atZone(ZoneId.systemDefault())
849//                                .toLocalDate();
850//                    } catch (ParseException e) {
851//                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + propertyType + " - Type read: " + dbValue.getClass() + " VALUE: " + dbValue;
852//                        throw new PersismException(msg, e);
853//                    }
854
855                } else if (propertyType.isEnum()) {
856                    // If this is an enum do a case insensitive comparison
857                    Object[] enumConstants = propertyType.getEnumConstants();
858                    for (Object element : enumConstants) {
859                        if (("" + dbValue).equalsIgnoreCase(element.toString())) {
860                            dbValue = element;
861                            break;
862                        }
863                    }
864                } else if (propertyType.equals(UUID.class)) {
865                    dbValue = UUID.fromString("" + dbValue);
866
867                } else if (propertyType.equals(Boolean.class) || propertyType.equals(boolean.class)) {
868                    // String to Boolean - true or 1 - otherwise false (or null)
869                    if (dbValue != null) {
870                        String bval = "" + dbValue;
871                        dbValue = (bval.equalsIgnoreCase("true") || bval.equals("1"));
872                    }
873
874                } else if (propertyType.equals(Time.class)) {
875                    // MSSQL works, JTDS returns Varchar in format below
876                    DateFormat df = new SimpleDateFormat("hh:mm:ss.SSSSS");
877                    try {
878                        Date d = df.parse(dbValue + "");
879                        dbValue = new Time(d.getTime());
880                    } catch (ParseException e) {
881                        String msg = e.getMessage() + ". Column: " + columnName + " Type of property: " + propertyType + " - Type read: " + dbValue.getClass() + " VALUE: " + dbValue;
882                        throw new PersismException(msg, e);
883                    }
884                } else if (propertyType.equals(Character.class) || propertyType.equals(char.class)) {
885                    String s = "" + dbValue;
886                    if (s.length() > 0) {
887                        dbValue = s.charAt(0);
888                    }
889                } else if (propertyType.equals(BigDecimal.class)) {
890                    dbValue = new BigDecimal("" + dbValue);
891                }
892                break;
893
894            case characterType:
895            case CharacterType:
896                // Does not occur because there's no direct single CHAR type.
897                // We get a string and handle it in the String case above.
898                log.debug("CharacterType");
899                break;
900
901            case UtilDateType:
902            case SQLDateType:
903            case TimestampType:
904                // todo maybe add check for property type = long. Same with TimestampType
905                if (propertyType.equals(java.util.Date.class) || propertyType.equals(java.sql.Date.class)) {
906                    if (propertyType.equals(java.sql.Date.class)) {
907                        dbValue = new java.sql.Date(((Date) dbValue).getTime());
908                    } else {
909                        dbValue = new java.util.Date(((Date) dbValue).getTime());
910                    }
911                }
912
913                if (propertyType.equals(LocalDate.class) || propertyType.equals(LocalDateTime.class)) {
914                    Date dt = (Date) dbValue;
915                    dbValue = Instant.ofEpochMilli(dt.getTime())
916                            .atZone(ZoneId.systemDefault())
917                            .toLocalDate();
918                }
919                break;
920
921            case TimeType:
922                log.debug("TimeType");
923                break;
924
925            case byteArrayType:
926            case ByteArrayType:
927                log.debug("ByteArrayType");
928                break;
929        }
930        return dbValue;
931    }

933    private <T> T getTypedValueReturnedFromGeneratedKeys(Class<T> type, ResultSet rs) throws SQLException {
934
935        Object value = null;
936        Types types = Types.getType(type);
937
938        if (types == null) {
939            log.warn("Unhandled type " + type);
940            return (T) rs.getObject(1);
941        }
942
943        switch (types) {
944
945            case integerType:
946            case IntegerType:
947                value = rs.getInt(1);
948                break;
949
950            case longType:
951            case LongType:
952                value = rs.getLong(1);
953                break;
954        }
955        return (T) value;
956    }

958    // Place code conversions here to prevent type exceptions on setObject
959    private static void setParameters(PreparedStatement st, Object[] parameters) throws SQLException {
960        if (log.isDebugEnabled()) {
961            log.debug("PARAMS: " + Arrays.asList(parameters));
962        }
963
964        int n = 1;
965        for (Object param : parameters) {
966
967            if (param != null) {
968
969                Types type;
970                if (param.getClass().isEnum()) {
971                    type = Types.EnumType;
972                } else {
973                    type = Types.getType(param.getClass());
974                }
975
976                assert type != null;
977
978                switch (type) {
979
980                    case booleanType:
981                    case BooleanType:
982                        st.setBoolean(n, (Boolean) param);
983                        break;
984
985                    case byteType:
986                    case ByteType:
987                        st.setByte(n, (Byte) param);
988                        break;
989
990                    case shortType:
991                    case ShortType:
992                        st.setShort(n, (Short) param);
993                        break;
994
995                    case integerType:
996                    case IntegerType:
997                        st.setInt(n, (Integer) param);
998                        break;
999
1000                    case longType:
1001                    case LongType:
1002                        st.setLong(n, (Long) param);
1003                        break;
1004
1005                    case floatType:
1006                    case FloatType:
1007                        st.setFloat(n, (Float) param);
1008                        break;
1009
1010                    case doubleType:
1011                    case DoubleType:
1012                        st.setDouble(n, (Double) param);
1013                        break;
1014
1015                    case DecimalType:
1016                        st.setBigDecimal(n, (BigDecimal) param);
1017                        break;
1018
1019                    case StringType:
1020                        st.setString(n, (String) param);
1021                        break;
1022
1023                    case characterType:
1024                    case CharacterType:
1025                        st.setObject(n, "" + param);
1026                        break;
1027
1028                    case UtilDateType:
1029                    case SQLDateType:
1030                        java.util.Date date = (Date) param;
1031                        st.setDate(n, new java.sql.Date(date.getTime()));
1032                        break;
1033
1034                    case TimeType:
1035                    case TimestampType:
1036                        // For Time we convert to Timestamp anyway
1037                        st.setTimestamp(n, (Timestamp) param);
1038                        break;
1039
1040                    case LocalDate:
1041                        LocalDate localDate = (LocalDate) param;
1042                        st.setTimestamp(n, Timestamp.valueOf(localDate.atStartOfDay()));
1043                        break;
1044
1045                    case LocalDateTime:
1046                        LocalDateTime ldt = (LocalDateTime) param;
1047                        st.setTimestamp(n, Timestamp.valueOf(ldt));
1048                        break;
1049
1050                    case Instant:
1051                        // todo Instant
1052                        break;
1053                    case OffsetDateTime:
1054                        // todo OffsetDateTime
1055                        break;
1056                    case ZonedDateTime:
1057                        // todo ZonedDateTime
1058                        break;
1059
1060                    case byteArrayType:
1061                    case ByteArrayType:
1062                        // Blob maps to byte array
1063                        st.setObject(n, param);
1064                        break;
1065
1066                    case ClobType:
1067                        log.debug("ClobType"); // doesn't occur since Clobs are mapped into Strings
1068                        st.setClob(n, (Clob) param);
1069                        break;
1070
1071                    case BlobType:
1072                        log.debug("BlobType"); // doesn't occur since Blobs are mapped into byte arrays
1073                        st.setBlob(n, (Blob) param);
1074                        break;
1075
1076                    case EnumType:
1077                        st.setString(n, param.toString());
1078                        break;
1079
1080                    case UUIDType:
1081                        st.setString(n, param.toString());
1082                        break;
1083                    default:
1084                        st.setObject(n, param);
1085                }
1086
1087            } else {
1088                // param is null
1089                st.setObject(n, param);
1090            }
1091
1092            n++;
1093        }
1094    }
