001package net.sf.persism;
002
003import java.lang.reflect.InvocationTargetException;
004import java.util.Collection;
005
006/**
007 * Indicates that Persism should keep track of changed columns in order to only include those columns in the SQL UPDATE statement.
008 * You can use this interface in situations where you can't or don't want your data objects to inherit PersistableObject.
009 *
010 * @author Dan Howard
011 * @see PersistableObject PersistableObject for implementation.
012 * @since 10/8/11 9:51 AM
013 */
014public interface Persistable<T> extends Cloneable {
015
016
017    /**
018     * Saves the current state of the data object to later detect changes for SQL UPDATE statements.
019     * Persism calls this method internally, you usually don't have to call this method yourself.
020     * @see PersistableObject for example implementation
021     * @throws PersismException If an SQL or other exception occurs.
022     */
023    void saveReadState() throws PersismException;
024
025    /**
026     * Getter for the data object in it's original state. The state at the time it was read from the database.
027     *
028     * @see PersistableObject for example implementation
029     * @return The data object in it's original state.
030     */
031    T getOriginalValue();
032}
