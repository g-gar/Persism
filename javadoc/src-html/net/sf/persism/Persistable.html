006    /**
007     * Interface for PersistableObject class. You can use this interface in situations where you can't or don't want
008     * your data objects to inherit PersistableObject.
009     * TODO Example how to implement.
010     *
011     * @author Dan Howard
012     * @see PersistableObject PersistableObject for implementation.
013     * @since 10/8/11 9:51 AM
014     */
015    public interface Persistable<T> extends Cloneable {
016    
017    
018        /**
019         * Saves the current state of the data object to later detect changes for SQL UPDATE statements.
020         * Persism calls this method internally, you usually don't have to call this method yourself.
021         * @see PersistableObject for example implementation
022         * @throws PersismException If an SQL or other exception occurs.
023         */
024        void saveReadState() throws PersismException;
025    
026        /**
027         * Getter for the data object in it's original state. The state at the time it was read from the database.
028         *
029         * @see PersistableObject for example implementation
030         * @return The data object in it's original state.
031         */
032        T getOriginalValue();
033    }
